109 int argon2_core(argon2_context *context, argon2_type type) {
110  /* 1. Validate all inputs */
111  int result = validate_inputs(context);
112  uint32_t memory_blocks, segment_length;
113  argon2_instance_t instance;
114 
115  if (ARGON2_OK != result) {
116  return result;
117  }
118 
119  if (Argon2_d != type && Argon2_i != type) {
120  return ARGON2_INCORRECT_TYPE;
121  }
122 
123  /* 2. Align memory size */
124  /* Minimum memory_blocks = 8L blocks, where L is the number of lanes */
125  memory_blocks = context->m_cost;
126 
127  if (memory_blocks < 2 * ARGON2_SYNC_POINTS * context->lanes) {
128  memory_blocks = 2 * ARGON2_SYNC_POINTS * context->lanes;
129  }
130 
131  segment_length = memory_blocks / (context->lanes * ARGON2_SYNC_POINTS);
132  /* Ensure that all segments have equal length */
133  memory_blocks = segment_length * (context->lanes * ARGON2_SYNC_POINTS);
134 
135  instance.memory = NULL;
136  instance.passes = context->t_cost;
137  instance.memory_blocks = memory_blocks;
138  instance.segment_length = segment_length;
139  instance.lane_length = segment_length * ARGON2_SYNC_POINTS;
140  instance.lanes = context->lanes;
141  instance.threads = context->threads;
142  instance.type = type;
143 
144  /* 3. Initialization: Hashing inputs, allocating memory, filling first
145  * blocks
146  */
147  result = initialize(&instance, context);
148 
149  if (ARGON2_OK != result) {
150  return result;
151  }
152 
153  /* 4. Filling memory */
154  result = fill_memory_blocks(&instance);
155 
156  if (ARGON2_OK != result) {
157  return result;
158  }
159  /* 5. Finalization */
160  finalize(context, &instance);
161 
162  return ARGON2_OK;
163 }

166 int argon2_hash(const uint32_t t_cost, const uint32_t m_cost,
167  const uint32_t parallelism, const void *pwd,
168  const size_t pwdlen, const void *salt, const size_t saltlen,
169  void *hash, const size_t hashlen, char *encoded,
170  const size_t encodedlen, argon2_type type) {
171 
172  argon2_context context;
173  int result;
174  uint8_t *out;
175 
176  /* Detect and reject overflowing sizes */
177  /* TODO: This should probably be fixed in the function signature */
178  if (pwdlen > UINT32_MAX) {
179  return ARGON2_PWD_TOO_LONG;
180  }
181 
182  if (hashlen > UINT32_MAX) {
183  return ARGON2_OUTPUT_TOO_LONG;
184  }
185 
186  if (saltlen > UINT32_MAX) {
187  return ARGON2_SALT_TOO_LONG;
188  }
189 
190  out = malloc(hashlen);
191  if (!out) {
192  return ARGON2_MEMORY_ALLOCATION_ERROR;
193  }
194 
195  context.out = (uint8_t *)out;
196  context.outlen = (uint32_t)hashlen;
197  context.pwd = CONST_CAST(uint8_t *)pwd;
198  context.pwdlen = (uint32_t)pwdlen;
199  context.salt = CONST_CAST(uint8_t *)salt;
200  context.saltlen = (uint32_t)saltlen;
201  context.secret = NULL;
202  context.secretlen = 0;
203  context.ad = NULL;
204  context.adlen = 0;
205  context.t_cost = t_cost;
206  context.m_cost = m_cost;
207  context.lanes = parallelism;
208  context.threads = parallelism;
209  context.allocate_cbk = NULL;
210  context.free_cbk = NULL;
211  context.flags = ARGON2_DEFAULT_FLAGS;
212 
213  result = argon2_core(&context, type);
214 
215  if (result != ARGON2_OK) {
216  memset(out, 0x00, hashlen);
217  free(out);
218  return result;
219  }
220 
221  /* if raw hash requested, write it */
222  if (hash) {
223  memcpy(hash, out, hashlen);
224  }
225 
226  /* if encoding requested, write it */
227  if (encoded && encodedlen) {
228  if (!encode_string(encoded, encodedlen, &context, type)) {
229  memset(out, 0x00, hashlen);
230  memset(encoded, 0x00, encodedlen);
231  free(out);
232  return ARGON2_ENCODING_FAIL;
233  }
234  }
235 
236  free(out);
237 
238  return ARGON2_OK;
239 }

279 static int argon2_compare(const uint8_t *b1, const uint8_t *b2, size_t len) {
280  size_t i;
281  uint8_t d = 0U;
282 
283  for (i = 0U; i < len; i++) {
284  d |= b1[i] ^ b2[i];
285  }
286  return (int)((1 & ((d - 1) >> 8)) - 1);
287 }

289 int argon2_verify(const char *encoded, const void *pwd, const size_t pwdlen,
290  argon2_type type) {
291 
292  argon2_context ctx;
293  uint8_t *out;
294  int ret;
295 
296  /* max values, to be updated in decode_string */
297  ctx.adlen = 512;
298  ctx.saltlen = 512;
299  ctx.outlen = 512;
300 
301  ctx.ad = malloc(ctx.adlen);
302  ctx.salt = malloc(ctx.saltlen);
303  ctx.out = malloc(ctx.outlen);
304  if (!ctx.out || !ctx.salt || !ctx.ad) {
305  free(ctx.ad);
306  free(ctx.salt);
307  free(ctx.out);
308  return ARGON2_MEMORY_ALLOCATION_ERROR;
309  }
310  out = malloc(ctx.outlen);
311  if (!out) {
312  free(ctx.ad);
313  free(ctx.salt);
314  free(ctx.out);
315  return ARGON2_MEMORY_ALLOCATION_ERROR;
316  }
317 
318  if(decode_string(&ctx, encoded, type) != 1) {
319  free(ctx.ad);
320  free(ctx.salt);
321  free(ctx.out);
322  free(out);
323  return ARGON2_DECODING_FAIL;
324  }
325 
326  ret = argon2_hash(ctx.t_cost, ctx.m_cost, ctx.threads, pwd, pwdlen, ctx.salt,
327  ctx.saltlen, out, ctx.outlen, NULL, 0, type);
328 
329  free(ctx.ad);
330  free(ctx.salt);
331 
332  if (ret != ARGON2_OK || argon2_compare(out, ctx.out, ctx.outlen)) {
333  free(out);
334  free(ctx.out);
335  return ARGON2_DECODING_FAIL;
336  }
337  free(out);
338  free(ctx.out);
339 
340  return ARGON2_OK;
341 }
