54 // select a block from the candidate blocks in vSortedByTimestamp, excluding
55 // already selected blocks in vSelectedBlocks, and with timestamp up to
56 // nSelectionIntervalStop.
57 static bool SelectBlockFromCandidates(vector<pair<int64_t, uint256> >& vSortedByTimestamp, map<uint256, const CBlockIndex*>& mapSelectedBlocks,
58  int64_t nSelectionIntervalStop, uint64_t nStakeModifierPrev, const CBlockIndex** pindexSelected)
59 {
60  bool fSelected = false;
61  uint256 hashBest = 0;
62  *pindexSelected = (const CBlockIndex*) 0;
63  BOOST_FOREACH(const PAIRTYPE(int64_t, uint256)& item, vSortedByTimestamp)
64  {
65  if (!mapBlockIndex.count(item.second))
66  return error("SelectBlockFromCandidates: failed to find block index for candidate block %s", item.second.ToString());
67  const CBlockIndex* pindex = mapBlockIndex[item.second];
68  if (fSelected && pindex->GetBlockTime() > nSelectionIntervalStop)
69  break;
70  if (mapSelectedBlocks.count(pindex->GetBlockHash()) > 0)
71  continue;
72  // compute the selection hash by hashing its proof-hash and the
73  // previous proof-of-stake modifier
74  CDataStream ss(SER_GETHASH, 0);
75  ss << pindex->hashProof << nStakeModifierPrev;
76  uint256 hashSelection = Hash(ss.begin(), ss.end());
77  // the selection hash is divided by 2**32 so that proof-of-stake block
78  // is always favored over proof-of-work block. this is to preserve
79  // the energy efficiency property
80  if (pindex->IsProofOfStake())
81  hashSelection >>= 32;
82  if (fSelected && hashSelection < hashBest)
83  {
84  hashBest = hashSelection;
85  *pindexSelected = (const CBlockIndex*) pindex;
86  }
87  else if (!fSelected)
88  {
89  fSelected = true;
90  hashBest = hashSelection;
91  *pindexSelected = (const CBlockIndex*) pindex;
92  }
93  }
94  LogPrint("stakemodifier", "SelectBlockFromCandidates: selection hash=%s\n", hashBest.ToString());
95  return fSelected;
96 }

98 // Stake Modifier (hash modifier of proof-of-stake):
99 // The purpose of stake modifier is to prevent a txout (coin) owner from
100 // computing future proof-of-stake generated by this txout at the time
101 // of transaction confirmation. To meet kernel protocol, the txout
102 // must hash with a future stake modifier to generate the proof.
103 // Stake modifier consists of bits each of which is contributed from a
104 // selected block of a given block group in the past.
105 // The selection of a block is based on a hash of the block's proof-hash and
106 // the previous stake modifier.
107 // Stake modifier is recomputed at a fixed time interval instead of every
108 // block. This is to make it difficult for an attacker to gain control of
109 // additional bits in the stake modifier, even after generating a chain of
110 // blocks.
111 bool ComputeNextStakeModifier(const CBlockIndex* pindexPrev, uint64_t& nStakeModifier, bool& fGeneratedStakeModifier)
112 {
113  nStakeModifier = 0;
114  fGeneratedStakeModifier = false;
115  if (!pindexPrev)
116  {
117  fGeneratedStakeModifier = true;
118  return true; // genesis block's modifier is 0
119  }
120  // First find current stake modifier and its generation block time
121  // if it's not old enough, return the same stake modifier
122  int64_t nModifierTime = 0;
123  if (!GetLastStakeModifier(pindexPrev, nStakeModifier, nModifierTime))
124  return error("ComputeNextStakeModifier: unable to get last modifier");
125  LogPrint("stakemodifier", "ComputeNextStakeModifier: prev modifier=0x%016x time=%s\n", nStakeModifier, DateTimeStrFormat(nModifierTime));
126  if (nModifierTime / nModifierInterval >= pindexPrev->GetBlockTime() / nModifierInterval)
127  return true;
128 
129  // Sort candidate blocks by timestamp
130  vector<pair<int64_t, uint256> > vSortedByTimestamp;
131  vSortedByTimestamp.reserve(64 * nModifierInterval / POS_TARGET_SPACING);
132  int64_t nSelectionInterval = GetStakeModifierSelectionInterval();
133  int64_t nSelectionIntervalStart = (pindexPrev->GetBlockTime() / nModifierInterval) * nModifierInterval - nSelectionInterval;
134  const CBlockIndex* pindex = pindexPrev;
135  while (pindex && pindex->GetBlockTime() >= nSelectionIntervalStart)
136  {
137  vSortedByTimestamp.push_back(make_pair(pindex->GetBlockTime(), pindex->GetBlockHash()));
138  pindex = pindex->pprev;
139  }
140  int nHeightFirstCandidate = pindex ? (pindex->nHeight + 1) : 0;
141  reverse(vSortedByTimestamp.begin(), vSortedByTimestamp.end());
142  sort(vSortedByTimestamp.begin(), vSortedByTimestamp.end());
143 
144  // Select 64 blocks from candidate blocks to generate stake modifier
145  uint64_t nStakeModifierNew = 0;
146  int64_t nSelectionIntervalStop = nSelectionIntervalStart;
147  map<uint256, const CBlockIndex*> mapSelectedBlocks;
148  for (int nRound=0; nRound<min(64, (int)vSortedByTimestamp.size()); nRound++)
149  {
150  // add an interval section to the current selection round
151  nSelectionIntervalStop += GetStakeModifierSelectionIntervalSection(nRound);
152  // select a block from the candidates of current round
153  if (!SelectBlockFromCandidates(vSortedByTimestamp, mapSelectedBlocks, nSelectionIntervalStop, nStakeModifier, &pindex))
154  return error("ComputeNextStakeModifier: unable to select block at round %d", nRound);
155  // write the entropy bit of the selected block
156  nStakeModifierNew |= (((uint64_t)pindex->GetStakeEntropyBit()) << nRound);
157  // add the selected block from candidates to selected list
158  mapSelectedBlocks.insert(make_pair(pindex->GetBlockHash(), pindex));
159  LogPrint("stakemodifier", "ComputeNextStakeModifier: selected round %d stop=%s height=%d bit=%d\n", nRound, DateTimeStrFormat(nSelectionIntervalStop), pindex->nHeight, pindex->GetStakeEntropyBit());
160  }
161 
162  // Print selection map for visualization of the selected blocks
163  if (LogAcceptCategory("stakemodifier"))
164  {
165  string strSelectionMap = "";
166  // '-' indicates proof-of-work blocks not selected
167  strSelectionMap.insert(0, pindexPrev->nHeight - nHeightFirstCandidate + 1, '-');
168  pindex = pindexPrev;
169  while (pindex && pindex->nHeight >= nHeightFirstCandidate)
170  {
171  // '=' indicates proof-of-stake blocks not selected
172  if (pindex->IsProofOfStake())
173  strSelectionMap.replace(pindex->nHeight - nHeightFirstCandidate, 1, "=");
174  pindex = pindex->pprev;
175  }
176  BOOST_FOREACH(const PAIRTYPE(uint256, const CBlockIndex*)& item, mapSelectedBlocks)
177  {
178  // 'S' indicates selected proof-of-stake blocks
179  // 'W' indicates selected proof-of-work blocks
180  strSelectionMap.replace(item.second->nHeight - nHeightFirstCandidate, 1, item.second->IsProofOfStake()? "S" : "W");
181  }
182  LogPrintf("ComputeNextStakeModifier: selection height [%d, %d] map %s\n", nHeightFirstCandidate, pindexPrev->nHeight, strSelectionMap);
183  }
184  LogPrint("stakemodifier", "ComputeNextStakeModifier: new modifier=0x%016x time=%s\n", nStakeModifierNew, DateTimeStrFormat(pindexPrev->GetBlockTime()));
185 
186  nStakeModifier = nStakeModifierNew;
187  fGeneratedStakeModifier = true;
188  return true;
189 }

206 // DarkSilk kernel protocol
207 // coinstake must meet hash target according to the protocol:
208 // kernel (input 0) must meet the formula
209 // hash(nStakeModifier + txPrev.block.nTime + txPrev.nTime + txPrev.vout.hash + txPrev.vout.n + nTime) < bnTarget * nWeight
210 // this ensures that the chance of getting a coinstake is proportional to the
211 // amount of coins one owns.
212 // The reason this hash is chosen is the following:
213 // nStakeModifier: scrambles computation to make it very difficult to precompute
214 // future proof-of-stake
215 // txPrev.block.nTime: prevent nodes from guessing a good timestamp to
216 // generate transaction for future advantage
217 // txPrev.nTime: slightly scrambles computation
218 // txPrev.vout.hash: hash of txPrev, to reduce the chance of nodes
219 // generating coinstake at the same time
220 // txPrev.vout.n: output number of txPrev, to reduce the chance of nodes
221 // generating coinstake at the same time
222 // nTime: current timestamp
223 // block/tx hash should not be used here as they can be generated in vast
224 // quantities so as to generate blocks faster, degrading the system back into
225 // a proof-of-work situation.
226 //
227 bool CheckStakeKernelHash(CBlockIndex* pindexPrev, unsigned int nBits, unsigned int nTimeBlockFrom, const CTransaction& txPrev, const COutPoint& prevout, unsigned int nTimeTx, uint256& hashProofOfStake, uint256& targetProofOfStake, bool fPrintProofOfStake)
228 {
229  if (nTimeTx < txPrev.nTime) // Transaction timestamp violation
230  return error("CheckStakeKernelHash() : nTime violation");
231 
232  // Base target
233  CBigNum bnTarget;
234  bnTarget.SetCompact(nBits);
235 
236  // Weighted target
237  int64_t nValueIn = txPrev.vout[prevout.n].nValue;
238  CBigNum bnWeight = CBigNum(nValueIn);
239  bnTarget *= bnWeight;
240 
241  targetProofOfStake = bnTarget.getuint256();
242 
243  uint64_t nStakeModifier = pindexPrev->nStakeModifier;
244  uint256 bnStakeModifierV2 = pindexPrev->bnStakeModifierV2;
245  int nStakeModifierHeight = pindexPrev->nHeight;
246  int64_t nStakeModifierTime = pindexPrev->nTime;
247 
248  // Calculate hash
249  CDataStream ss(SER_GETHASH, 0);
250  ss << bnStakeModifierV2;
251  ss << txPrev.nTime << prevout.hash << prevout.n << nTimeTx;
252  hashProofOfStake = Hash(ss.begin(), ss.end());
253 
254  if (fPrintProofOfStake)
255  {
256  LogPrintf("CheckStakeKernelHash() : using modifier 0x%016x at height=%d timestamp=%s for block from timestamp=%s\n",
257  nStakeModifier, nStakeModifierHeight,
258  DateTimeStrFormat(nStakeModifierTime),
259  DateTimeStrFormat(nTimeBlockFrom));
260  LogPrintf("CheckStakeKernelHash() : check modifier=0x%016x nTimeBlockFrom=%u nTimeTxPrev=%u nPrevout=%u nTimeTx=%u hashProof=%s\n",
261  nStakeModifier,
262  nTimeBlockFrom, txPrev.nTime, prevout.n, nTimeTx,
263  hashProofOfStake.ToString());
264  }
265 
266  // Now check if proof-of-stake hash meets target protocol
267  if (CBigNum(hashProofOfStake) > bnTarget)
268  return false;
269 
270  if (fDebug && !fPrintProofOfStake)
271  {
272  LogPrintf("CheckStakeKernelHash() : using modifier 0x%016x at height=%d timestamp=%s for block from timestamp=%s\n",
273  nStakeModifier, nStakeModifierHeight,
274  DateTimeStrFormat(nStakeModifierTime),
275  DateTimeStrFormat(nTimeBlockFrom));
276  LogPrintf("CheckStakeKernelHash() : pass modifier=0x%016x nTimeBlockFrom=%u nTimeTxPrev=%u nPrevout=%u nTimeTx=%u hashProof=%s\n",
277  nStakeModifier,
278  nTimeBlockFrom, txPrev.nTime, prevout.n, nTimeTx,
279  hashProofOfStake.ToString());
280  }
281 
282  return true;
283 }

309  return fDebug? error("CheckProofOfStake() : read block failed") : false; // unable to read block of previous transaction

312

318 
319  return true;
320 }
