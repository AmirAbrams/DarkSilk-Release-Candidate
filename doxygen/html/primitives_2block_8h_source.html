196  uint256 BuildMerkleTree() const
197  {
198  vMerkleTree.clear();
199  BOOST_FOREACH(const CTransaction& tx, vtx)
200  vMerkleTree.push_back(tx.GetHash());
201  int j = 0;
202  for (int nSize = vtx.size(); nSize > 1; nSize = (nSize + 1) / 2)
203  {
204  for (int i = 0; i < nSize; i += 2)
205  {
206  int i2 = std::min(i+1, nSize-1);
207  vMerkleTree.push_back(Hash(BEGIN(vMerkleTree[j+i]), END(vMerkleTree[j+i]),
208  BEGIN(vMerkleTree[j+i2]), END(vMerkleTree[j+i2])));
209  }
210  j += nSize;
211  }
212  return (vMerkleTree.empty() ? 0 : vMerkleTree.back());
213  }
