77 bool parseCommandLine(std::vector<std::string> &args, const std::string &strCommand)
78 {
79  enum CmdParseState
80  {
81  STATE_EATING_SPACES,
82  STATE_ARGUMENT,
83  STATE_SINGLEQUOTED,
84  STATE_DOUBLEQUOTED,
85  STATE_ESCAPE_OUTER,
86  STATE_ESCAPE_DOUBLEQUOTED
87  } state = STATE_EATING_SPACES;
88  std::string curarg;
89  foreach(char ch, strCommand)
90  {
91  switch(state)
92  {
93  case STATE_ARGUMENT: // In or after argument
94  case STATE_EATING_SPACES: // Handle runs of whitespace
95  switch(ch)
96  {
97  case '"': state = STATE_DOUBLEQUOTED; break;
98  case '\'': state = STATE_SINGLEQUOTED; break;
99  case '\\': state = STATE_ESCAPE_OUTER; break;
100  case ' ': case '\n': case '\t':
101  if(state == STATE_ARGUMENT) // Space ends argument
102  {
103  args.push_back(curarg);
104  curarg.clear();
105  }
106  state = STATE_EATING_SPACES;
107  break;
108  default: curarg += ch; state = STATE_ARGUMENT;
109  }
110  break;
111  case STATE_SINGLEQUOTED: // Single-quoted string
112  switch(ch)
113  {
114  case '\'': state = STATE_ARGUMENT; break;
115  default: curarg += ch;
116  }
117  break;
118  case STATE_DOUBLEQUOTED: // Double-quoted string
119  switch(ch)
120  {
121  case '"': state = STATE_ARGUMENT; break;
122  case '\\': state = STATE_ESCAPE_DOUBLEQUOTED; break;
123  default: curarg += ch;
124  }
125  break;
126  case STATE_ESCAPE_OUTER: // '\' outside quotes
127  curarg += ch; state = STATE_ARGUMENT;
128  break;
129  case STATE_ESCAPE_DOUBLEQUOTED: // '\' in double-quoted text
130  if(ch != '"' && ch != '\\') curarg += '\\'; // keep '\' for everything but the quote and '\' itself
131  curarg += ch; state = STATE_DOUBLEQUOTED;
132  break;
133  }
134  }
135  switch(state) // final state
136  {
137  case STATE_EATING_SPACES:
138  return true;
139  case STATE_ARGUMENT:
140  args.push_back(curarg);
141  return true;
142  default: // ERROR to end in one of the other states
143  return false;
144  }
145 }

154 void RPCExecutor::request(const QString &command)
155 {
156  std::vector<std::string> args;
157  if(!parseCommandLine(args, command.toStdString()))
158  {
159  emit reply(DEBUGConsole::CMD_ERROR, QString("Parse error: unbalanced ' or \""));
160  return;
161  }
162  if(args.empty())
163  return; // Nothing to do
164  try
165  {
166  std::string strPrint;
167  // Convert argument list to JSON objects in method-dependent way,
168  // and pass it along with the method name to the dispatcher.
169  json_spirit::Value result = tableRPC.execute(
170  args[0],
171  RPCConvertValues(args[0], std::vector<std::string>(args.begin() + 1, args.end())));
172 
173  // Format result reply
174  if (result.type() == json_spirit::null_type)
175  strPrint = "";
176  else if (result.type() == json_spirit::str_type)
177  strPrint = result.get_str();
178  else

182  }

192  { // Show raw JSON object
193  emit reply(DEBUGConsole::CMD_ERROR, QString::fromStdString(write_string(json_spirit::Value(objError), false)));
194  }

578 void DEBUGConsole::startExecutor()
579 {
580  QThread* thread = new QThread;
581  RPCExecutor *executor = new RPCExecutor();
582  executor->moveToThread(thread);
583 
584  // Notify executor when thread started (in executor thread)
585  connect(thread, SIGNAL(started()), executor, SLOT(start()));
586  // Replies from executor object must go to this object
587  connect(executor, SIGNAL(reply(int,QString)), this, SLOT(message(int,QString)));
588  // Requests from this object must go to executor
589  connect(this, SIGNAL(cmdRequest(QString)), executor, SLOT(request(QString)));
590  // On stopExecutor signal
591  // - queue executor for deletion (in execution thread)
592  // - quit the Qt event loop in the execution thread
593  connect(this, SIGNAL(stopExecutor()), executor, SLOT(deleteLater()));
594  connect(this, SIGNAL(stopExecutor()), thread, SLOT(quit()));
595  // Queue the thread for deletion (in this thread) when it is finished
596  connect(thread, SIGNAL(finished()), thread, SLOT(deleteLater()));
597 
598  // Default implementation of QThread::run() simply spins up an event loop in the thread,
599  // which is what we want.
600  thread->start();
601 }
