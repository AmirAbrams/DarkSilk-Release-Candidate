501 bool AddOrphanTx(const CTransaction& tx, NodeId peer)
502 {
503  uint256 hash = tx.GetHash();
504  if (mapOrphanTransactions.count(hash))
505  return false;
506 
507  // Ignore big transactions, to avoid a
508  // send-big-orphans memory exhaustion attack. If a peer has a legitimate
509  // large transaction with a missing parent then we assume
510  // it will rebroadcast it later, after the parent transaction(s)
511  // have been mined or received.
512  // 10,000 orphans, each of which is at most 5,000 bytes big is
513  // at most 500 megabytes of orphans:
514  unsigned int sz = tx.GetSerializeSize(SER_NETWORK, CTransaction::CURRENT_VERSION);
515  if (sz > 5000)
516  {
517  LogPrint("mempool", "ignoring large orphan tx (size: %u, hash: %s)\n", sz, hash.ToString());
518  return false;
519  }
520 
521  mapOrphanTransactions[hash].tx = tx;
522  mapOrphanTransactions[hash].fromPeer = peer;
523  BOOST_FOREACH(const CTxIn& txin, tx.vin)
524  mapOrphanTransactionsByPrev[txin.prevout.hash].insert(hash);
525 
526  LogPrint("mempool", "stored orphan tx %s (mapsz %u prevsz %u)\n", hash.ToString(),
527  mapOrphanTransactions.size(), mapOrphanTransactionsByPrev.size());
528  return true;
529 }

569 bool IsStandardTx(const CTransaction& tx, string& reason)
570 {
571  if (tx.nVersion > CTransaction::CURRENT_VERSION || tx.nVersion < 1) {
572  reason = "version";
573  return false;
574  }
575 
576  // Treat non-final transactions as non-standard to prevent a specific type
577  // of double-spend attack, as well as DoS attacks. (if the transaction
578  // can't be mined, the attacker isn't expending resources broadcasting it)
579  // Basically we don't want to propagate transactions that can't be included in
580  // the next block.
581  //
582  // However, IsFinalTx() is confusing... Without arguments, it uses
583  // nBestHeight to evaluate nLockTime; when a block is accepted, nBestHeight
584  // is set to the value of nHeight in the block. However, when IsFinalTx()
585  // is called within CBlock::AcceptBlock(), the height of the block *being*
586  // evaluated is what is used. Thus if we want to know if a transaction can
587  // be part of the *next* block, we need to call IsFinalTx() with one more
588  // than nBestHeight.
589  //
590  // Timestamps on the other hand don't get any special treatment, because we
591  // can't know what timestamp the next block will have, and there aren't
592  // timestamp applications where it matters.
593  if (!IsFinalTx(tx, nBestHeight + 1)) {
594  reason = "non-final";
595  return false;
596  }
597  // nTime has different purpose from nLockTime but can be used in similar attacks
598  if (tx.nTime > FutureDrift(GetAdjustedTime(), true)) {
599  reason = "time-too-new";
600  return false;
601  }
602 
603  // Extremely large transactions with lots of inputs can cost the network
604  // almost as much to process as they cost the sender in fees, because
605  // computing signature hashes is O(ninputs*txsize). Limiting transactions
606  // to MAX_STANDARD_TX_SIZE mitigates CPU exhaustion attacks.
607  unsigned int sz = tx.GetSerializeSize(SER_NETWORK, CTransaction::CURRENT_VERSION);
608  if (sz >= MAX_STANDARD_TX_SIZE) {
609  reason = "tx-size";
610  return false;
611  }
612 
613  BOOST_FOREACH(const CTxIn& txin, tx.vin)
614  {
615  // Biggest 'standard' txin is a 15-of-15 P2SH multisig with compressed
616  // keys. (remember the 520 byte limit on redeemScript size) That works
617  // out to a (15*(33+1))+3=513 byte redeemScript, 513+1+15*(73+1)+3=1627
618  // bytes of scriptSig, which we round off to 1650 bytes for some minor
619  // future-proofing. That's also enough to spend a 20-of-20
620  // CHECKMULTISIG scriptPubKey, though such a scriptPubKey is not
621  // considered standard)
622  if (txin.scriptSig.size() > 1650) {
623  reason = "scriptsig-size";
624  return false;
625  }
626  if (!txin.scriptSig.IsPushOnly()) {
627  reason = "scriptsig-not-pushonly";
628  return false;
629  }
630  if (!txin.scriptSig.HasCanonicalPushes()) {
631  reason = "scriptsig-non-canonical-push";
632  return false;
633  }
634  }
635 
636  unsigned int nDataOut = 0;
637  txnouttype whichType;
638  BOOST_FOREACH(const CTxOut& txout, tx.vout) {
639  if (!::IsStandard(txout.scriptPubKey, whichType)) {
640  reason = "scriptpubkey";
641  return false;
642  }
643  if (whichType == TX_NULL_DATA)
644  nDataOut++;
645  else if (txout.nValue == 0) {
646  reason = "dust";
647  return false;
648  }
649  if (!txout.scriptPubKey.HasCanonicalPushes()) {
650  reason = "scriptpubkey-non-canonical-push";
651  return false;
652  }
653  }
654 
655  // not more than one data txout per non-data txout is permitted
656  // only one data txout is permitted too
657  if (nDataOut > 1 && nDataOut > tx.vout.size()/2) {
658  reason = "multi-op-return";
659  return false;
660  }
661 
662  return true;
663 }

685 // potential denial-of-service attacks:
686 //
687 // 1. scriptSigs with extra data stuffed into them,
688 // not consumed by scriptPubKey (or P2SH script)
689 // 2. P2SH scripts with a crazy number of expensive
690 // CHECKSIG/CHECKMULTISIG operations
691 //
692 bool AreInputsStandard(const CTransaction& tx, const MapPrevTx& mapInputs)
693 {
694  if (tx.IsCoinBase())
695  return true; // Coinbases don't use vin normally
696 
697  for (unsigned int i = 0; i < tx.vin.size(); i++)
698  {
699  CTransactionPoS txPoS;
700  const CTxOut& prev = txPoS.GetOutputFor(tx.vin[i], mapInputs);
701 
702  vector<vector<unsigned char> > vSolutions;
703  txnouttype whichType;
704  // get the scriptPubKey corresponding to this input:
705  const CScript& prevScript = prev.scriptPubKey;
706  if (!Solver(prevScript, whichType, vSolutions))
707  return false;
708  int nArgsExpected = ScriptSigArgsExpected(whichType, vSolutions);
709  if (nArgsExpected < 0)
710  return false;
711 
712  // Transactions with extra stuff in their scriptSigs are
713  // non-standard. Note that this EvalScript() call will
714  // be quick, because if there are any operations
715  // beside "push data" in the scriptSig
716  // IsStandard() will have already returned false
717  // and this method isn't called.
718  vector<vector<unsigned char> > stack;
719  if (!EvalScript(stack, tx.vin[i].scriptSig, tx, i, SCRIPT_VERIFY_NONE, 0))
720  return false;
721 
722  if (whichType == TX_SCRIPTHASH)
723  {
724  if (stack.empty())
725  return false;
726  CScript subscript(stack.back().begin(), stack.back().end());
727  vector<vector<unsigned char> > vSolutions2;
728  txnouttype whichType2;
729  if (Solver(subscript, whichType2, vSolutions2))
730  {
731  int tmpExpected = ScriptSigArgsExpected(whichType2, vSolutions2);
732 
733  if (whichType2 == TX_SCRIPTHASH)
734  return false;
735  if (tmpExpected < 0)
736  return false;
737 
738  nArgsExpected += tmpExpected;
739  }
740  else
741  {
742  // Any other Script with less than 15 sigops OK:
743  unsigned int sigops = subscript.GetSigOpCount(true);
744  // ... extra data left on the stack after execution is OK, too:
745  return (sigops <= MAX_P2SH_SIGOPS);
746  }
747  }
748 
749  if (stack.size() != (unsigned int)nArgsExpected)
750  return false;
751  }
752 
753  return true;
754 }
