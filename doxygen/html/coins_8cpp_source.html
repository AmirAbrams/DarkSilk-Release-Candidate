153 bool CCoinsViewCache::BatchWrite(CCoinsMap &mapCoins, const uint256 &hashBlockIn) {
154  assert(!hasModifier);
155  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
156  if (it->second.flags & CCoinsCacheEntry::DIRTY) { // Ignore non-dirty entries (optimization).
157  CCoinsMap::iterator itUs = cacheCoins.find(it->first);
158  if (itUs == cacheCoins.end()) {
159  if (!it->second.coins.IsPruned()) {
160  // The parent cache does not have an entry, while the child
161  // cache does have (a non-pruned) one. Move the data up, and
162  // mark it as fresh (if the grandparent did have it, we
163  // would have pulled it in at first GetCoins).
164  assert(it->second.flags & CCoinsCacheEntry::FRESH);
165  CCoinsCacheEntry& entry = cacheCoins[it->first];
166  entry.coins.swap(it->second.coins);
167  entry.flags = CCoinsCacheEntry::DIRTY | CCoinsCacheEntry::FRESH;
168  }
169  } else {
170  if ((itUs->second.flags & CCoinsCacheEntry::FRESH) && it->second.coins.IsPruned()) {
171  // The grandparent does not have an entry, and the child is
172  // modified and being pruned. This means we can just delete
173  // it from the parent.
174  cacheCoins.erase(itUs);
175  } else {
176  // A normal modification.
177  itUs->second.coins.swap(it->second.coins);
178  itUs->second.flags |= CCoinsCacheEntry::DIRTY;
179  }
180  }
181  }
182  CCoinsMap::iterator itOld = it++;
183  mapCoins.erase(itOld);
184  }
185  hashBlock = hashBlockIn;
186  return true;
187 }

302 bool CCoinsViewDB::GetStats(CCoinsStats &stats) const {
303  /* It seems that there are no "const iterators" for LevelDB. Since we
304  only need read operations on it, use a const-cast to get around
305  that restriction. */
306  boost::scoped_ptr<leveldb::Iterator> pcursor(const_cast<CLevelDBWrapper*>(&db)->NewIterator());
307  pcursor->SeekToFirst();
308 
309  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
310  stats.hashBlock = GetBestBlock();
311  ss << stats.hashBlock;
312  CAmount nTotalAmount = 0;
313  while (pcursor->Valid()) {
314  boost::this_thread::interruption_point();
315  try {
316  leveldb::Slice slKey = pcursor->key();
317  CDataStream ssKey(slKey.data(), slKey.data()+slKey.size(), SER_DISK, CLIENT_VERSION);
318  char chType;
319  ssKey >> chType;
320  if (chType == 'c') {
321  leveldb::Slice slValue = pcursor->value();
322  CDataStream ssValue(slValue.data(), slValue.data()+slValue.size(), SER_DISK, CLIENT_VERSION);
323  CCoins coins;
324  ssValue >> coins;
325  uint256 txhash;
326  ssKey >> txhash;
327  ss << txhash;
328  ss << VARINT(coins.nVersion);
329  ss << (coins.fCoinBase ? 'c' : 'n');
330  ss << VARINT(coins.nHeight);
331  stats.nTransactions++;
332  for (unsigned int i=0; i<coins.vout.size(); i++) {
333  const CTxOut &out = coins.vout[i];
334  if (!out.IsNull()) {
335  stats.nTransactionOutputs++;
336  ss << VARINT(i+1);
337  ss << out;
338  nTotalAmount += out.nValue;
339  }
340  }
341  stats.nSerializedSize += 32 + slValue.size();
342  ss << VARINT(0);
343  }
344  pcursor->Next();
345  } catch (std::exception &e) {
346  return error("%s : Deserialize or I/O error - %s", __func__, e.what());
347  }
348  }
349  stats.nHeight = mapBlockIndex.find(GetBestBlock())->second->nHeight;
350  stats.hashSerialized = ss.GetHash();
351  stats.nTotalAmount = nTotalAmount;
352  return true;
353 }

373 // Amount compression:
374 // * If the amount is 0, output 0
375 // * first, divide the amount (in base units) by the largest power of 10 possible; call the exponent e (e is max 9)
376 // * if e<9, the last digit of the resulting number cannot be 0; store it as d, and drop it (divide by 10)
377 // * call the result n
378 // * output 1 + 10*(9*n + d - 1) + e
379 // * if e==9, we only know the resulting number is not zero, so output 1 + 10*(n - 1) + 9
380 // (this is decodable, as d is in [1-9] and e is in [0-9])
381 
382 uint64_t CTxOutCompressorCoin::CompressAmount(uint64_t n)
383 {
384  if (n == 0)
385  return 0;
386  int e = 0;
387  while (((n % 10) == 0) && e < 9) {
388  n /= 10;
389  e++;
390  }
391  if (e < 9) {
392  int d = (n % 10);
393  assert(d >= 1 && d <= 9);
394  n /= 10;
395  return 1 + (n*9 + d - 1)*10 + e;
396  } else {
397  return 1 + (n - 1)*10 + 9;
398  }
399 }

401 uint64_t CTxOutCompressorCoin::DecompressAmount(uint64_t x)
402 {
403  // x = 0 OR x = 1+10*(9*n + d - 1) + e OR x = 1+10*(n - 1) + 9
404  if (x == 0)
405  return 0;
406  x--;
407  // x = 10*(9*n + d - 1) + e
408  int e = x % 10;
409  x /= 10;
410  uint64_t n = 0;
411  if (e < 9) {
412  // x = 9*n + d - 1
413  int d = (x % 9) + 1;
414  x /= 9;
415  // x = n
416  n = x*10 + d;
417  } else {
418  n = x+1;
419  }
420  while (e) {
421  n *= 10;
422  e--;
423  }
424  return n;
425 }
