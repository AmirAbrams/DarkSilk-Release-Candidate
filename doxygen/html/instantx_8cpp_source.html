175 bool IsIXTXValid(const CTransaction& txCollateral){
176  if(txCollateral.vout.size() < 1) return false;
177  if(txCollateral.nLockTime != 0) return false;
178 
179  CAmount nValueIn = 0;
180  CAmount nValueOut = 0;
181  bool missingTx = false;
182 
183  BOOST_FOREACH(const CTxOut o, txCollateral.vout)
184  nValueOut += o.nValue;
185 
186  BOOST_FOREACH(const CTxIn i, txCollateral.vin){
187  CTransaction tx2;
188  uint256 hash;
189  if(GetTransaction(i.prevout.hash, tx2, hash)){
190  if(tx2.vout.size() > i.prevout.n) {
191  nValueIn += tx2.vout[i.prevout.n].nValue;
192  }
193  } else{
194  missingTx = true;
195  }
196  }
197 
198  if(nValueOut > GetSporkValue(SPORK_5_MAX_VALUE)*COIN){
199  LogPrint("instantx", "IsIXTXValid - Transaction value too high - %s\n", txCollateral.ToString().c_str());
200  return false;
201  }
202 
203  if(missingTx){
204  LogPrint("instantx", "IsIXTXValid - Unknown inputs in IX transaction - %s\n", txCollateral.ToString().c_str());
205  /*
206  This happens sometimes for an unknown reason, so we'll return that it's a valid transaction.
207  If someone submits an invalid transaction it will be rejected by the network anyway and this isn't
208  very common, but we don't want to block IX just because the client can't figure out the fee.
209  */
210  return true;
211  }
212 
213  if(nValueIn-nValueOut < CENT) {
214  LogPrint("instantx", "IsIXTXValid - did not include enough fees in transaction %d\n%s\n", nValueOut-nValueIn, txCollateral.ToString().c_str());
215  return false;
216  }
217 
218  return true;

221 int64_t CreateNewLock(CTransaction tx)
222 {
223 
224  int64_t nTxAge = 0;
225  BOOST_REVERSE_FOREACH(CTxIn i, tx.vin){
226  nTxAge = GetInputAge(i);
227  if(nTxAge < 9) //9 confirmations, 1 less than the "send IX" gui requires, incase of a block propagating the network at the time
228  {
229  LogPrintf("CreateNewLock - Transaction not found / too new: %d / %s\n", nTxAge, tx.GetHash().ToString().c_str());
230  return 0;
231  }
232  }
233 
234  /*
235  Use a blockheight newer than the input.
236  This prevents attackers from using transaction mallibility to predict which stormnodes
237  they'll use.
238  */
239  int nBlockHeight = (pindexBest->nHeight - nTxAge)+4;
240 
241  if (!mapTxLocks.count(tx.GetHash())){
242  LogPrintf("CreateNewLock - New Transaction Lock %s !\n", tx.GetHash().ToString().c_str());
243 
244  CTransactionLock newLock;
245  newLock.nBlockHeight = nBlockHeight;
246  newLock.nExpiration = GetTime()+(60*60); //locks expire after 60 minutes (56-60 confirmations)
247  newLock.nTimeout = GetTime()+(60*5);
248  newLock.txHash = tx.GetHash();
249  mapTxLocks.insert(make_pair(tx.GetHash(), newLock));
250  } else {
251  mapTxLocks[tx.GetHash()].nBlockHeight = nBlockHeight;
252  LogPrint("instantx", "CreateNewLock - Transaction Lock Exists %s !\n", tx.GetHash().ToString().c_str());
253  }
254 
255  return nBlockHeight;

396 bool CheckForConflictingLocks(CTransaction& tx)
397 {
398  /*
399  It's possible (very unlikely though) to get 2 conflicting transaction locks approved by the network.
400  In that case, they will cancel each other out.
401 
402  Blocks could have been rejected during this time, which is OK. After they cancel out, the client will
403  rescan the blocks and find they're acceptable and then take the chain with the most work.
404  */
405  BOOST_FOREACH(const CTxIn& in, tx.vin){
406  if(mapLockedInputs.count(in.prevout)){
407  if(mapLockedInputs[in.prevout] != tx.GetHash()){
408  LogPrintf("InstantX::CheckForConflictingLocks - found two complete conflicting locks - removing both. %s %s", tx.GetHash().ToString().c_str(), mapLockedInputs[in.prevout].ToString().c_str());
409  if(mapTxLocks.count(tx.GetHash())) mapTxLocks[tx.GetHash()].nExpiration = GetTime();
410  if(mapTxLocks.count(mapLockedInputs[in.prevout])) mapTxLocks[mapLockedInputs[in.prevout]].nExpiration = GetTime();
411  return true;
412  }
413  }
414  }
415 
416  return false;
417 }
