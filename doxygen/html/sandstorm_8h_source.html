106  bool Add(const std::vector<CTxIn> vinIn, CAmount amountIn, const CTransaction collateralIn, const std::vector<CTxOut> voutIn)
107  {
108  if(isSet){return false;}
109 
110  BOOST_FOREACH(const CTxIn& in, vinIn)
111  sev.push_back(in);
112 
113  BOOST_FOREACH(const CTxOut& out, voutIn)
114  vout.push_back(out);
115 
116  amount = amountIn;
117  collateral = collateralIn;
118  isSet = true;
119  addedTime = GetTime();
120 
121  return true;
122  }
123 
124  bool AddSig(const CTxIn& vin)
125  {
126  BOOST_FOREACH(CTxSSIn& s, sev) {
127  if(s.prevout == vin.prevout && s.nSequence == vin.nSequence){
128  if(s.fHasSig){return false;}
129  s.scriptSig = vin.scriptSig;
130  s.prevPubKey = vin.prevPubKey;
131  s.fHasSig = true;
132 
133  return true;
134  }
135  }
136 
137  return false;
138  }
139 
140  bool IsExpired()
141  {
142  return (GetTime() - addedTime) > SANDSTORM_QUEUE_TIMEOUT;// 120 seconds
