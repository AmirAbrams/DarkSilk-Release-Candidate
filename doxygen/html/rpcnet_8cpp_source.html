162  {
163  LOCK(cs_vAddedNodes);
164  BOOST_FOREACH(string& strAddNode, vAddedNodes)
165  laddedNodes.push_back(strAddNode);
166  }
167  else
168  {

171  BOOST_FOREACH(string& strAddNode, vAddedNodes)
172  if (strAddNode == strNode)
173  {
174  laddedNodes.push_back(strAddNode);
175  break;
176  }

178  throw JSONRPCError(RPC_CLIENT_NODE_NOT_ADDED, "Error: Node has not been added.");
179  }
180 
181  if (!fDns)
182  {
183  Object ret;
184  BOOST_FOREACH(string& strAddNode, laddedNodes)
185  ret.push_back(Pair("addednode", strAddNode));
186  return ret;
187  }
188 
189  Array ret;
190 
191  list<pair<string, vector<CService> > > laddedAddreses(0);
192  BOOST_FOREACH(string& strAddNode, laddedNodes)
193  {
194  vector<CService> vservNode(0);
195  if(Lookup(strAddNode.c_str(), vservNode, Params().GetDefaultPort(), fNameLookup, 0))
196  laddedAddreses.push_back(make_pair(strAddNode, vservNode));
197  else
198  {
199  Object obj;
200  obj.push_back(Pair("addednode", strAddNode));
201  obj.push_back(Pair("connected", false));

203  obj.push_back(Pair("addresses", addresses));
204  }
205  }
206

209  {
210  Object obj;
211  obj.push_back(Pair("addednode", it->first));
212 
213  Array addresses;
214  bool fConnected = false;
215  BOOST_FOREACH(CService& addrNode, it->second)
216  {
217  bool fFound = false;
218  Object node;
219  node.push_back(Pair("address", addrNode.ToString()));

221  if (pnode->addr == addrNode)
222  {
223  fFound = true;
224  fConnected = true;
225  node.push_back(Pair("connected", pnode->fInbound ? "inbound" : "outbound"));
226  break;
227  }
228  if (!fFound)
229  node.push_back(Pair("connected", "false"));
230  addresses.push_back(node);
231  }
232  obj.push_back(Pair("connected", fConnected));
233  obj.push_back(Pair("addresses", addresses));
234  ret.push_back(obj);
235  }
236 
237  return ret;
238 }

252  "<maxver> is the maximum applicable internal client version\n"
253  "<priority> is integer priority number\n"
254  "<id> is the alert id\n"
255  "[cancelupto] cancels all alert id's up to this number\n"
256  "Returns true or false.");

260

272  CDataStream sMsg(SER_NETWORK, PROTOCOL_VERSION);
273  sMsg << (CUnsignedAlert)alert;
274  alert.vchMsg = vector<unsigned char>(sMsg.begin(), sMsg.end());
275 
276  vector<unsigned char> vchPrivKey = ParseHex(params[1].get_str());
277  key.SetPrivKey(CPrivKey(vchPrivKey.begin(), vchPrivKey.end()), false); // if key is not correct openssl may crash
