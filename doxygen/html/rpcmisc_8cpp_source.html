218  CDarkSilkAddress addr(strAddress);

224  throw JSONRPCError(RPC_TYPE_ERROR, "Address does not refer to key");
225 
226  bool fInvalid = false;
227  vector<unsigned char> vchSig = DecodeBase64(strSign.c_str(), &fInvalid);

229  if (fInvalid)

232  CHashWriter ss(SER_GETHASH, 0);

238  return false;
