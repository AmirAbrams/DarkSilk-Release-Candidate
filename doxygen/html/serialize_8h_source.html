176 //
177 // Compact size
178 // size < 253 -- 1 byte
179 // size <= USHRT_MAX -- 3 bytes (253 + 2 bytes)
180 // size <= UINT_MAX -- 5 bytes (254 + 4 bytes)
181 // size > UINT_MAX -- 9 bytes (255 + 8 bytes)
182 //
183 inline unsigned int GetSizeOfCompactSize(uint64_t nSize)
184 {
185  if (nSize < 253) return sizeof(unsigned char);
186  else if (nSize <= std::numeric_limits<unsigned short>::max()) return sizeof(unsigned char) + sizeof(unsigned short);
187  else if (nSize <= std::numeric_limits<unsigned int>::max()) return sizeof(unsigned char) + sizeof(unsigned int);
188  else return sizeof(unsigned char) + sizeof(uint64_t);
189 }

191 template<typename Stream>
192 void WriteCompactSize(Stream& os, uint64_t nSize)
193 {
194  if (nSize < 253)
195  {
196  unsigned char chSize = nSize;
197  WRITEDATA(os, chSize);
198  }
199  else if (nSize <= std::numeric_limits<unsigned short>::max())
200  {
201  unsigned char chSize = 253;
202  unsigned short xSize = nSize;
203  WRITEDATA(os, chSize);
204  WRITEDATA(os, xSize);
205  }
206  else if (nSize <= std::numeric_limits<unsigned int>::max())
207  {
208  unsigned char chSize = 254;
209  unsigned int xSize = nSize;
210  WRITEDATA(os, chSize);
211  WRITEDATA(os, xSize);
212  }
213  else
214  {
215  unsigned char chSize = 255;
216  uint64_t xSize = nSize;
217  WRITEDATA(os, chSize);
218  WRITEDATA(os, xSize);
219  }
220  return;
221 }

223 template<typename Stream>
224 uint64_t ReadCompactSize(Stream& is)
225 {
226  unsigned char chSize;
227  READDATA(is, chSize);
228  uint64_t nSizeRet = 0;
229  if (chSize < 253)
230  {
231  nSizeRet = chSize;
232  }
233  else if (chSize == 253)
234  {
235  unsigned short xSize;
236  READDATA(is, xSize);
237  nSizeRet = xSize;
238  if (nSizeRet < 253)
239  throw std::ios_base::failure("non-canonical ReadCompactSize()");
240  }
241  else if (chSize == 254)
242  {
243  unsigned int xSize;
244  READDATA(is, xSize);
245  nSizeRet = xSize;
246  if (nSizeRet < 0x10000u)
247  throw std::ios_base::failure("non-canonical ReadCompactSize()");
248  }
249  else
250  {
251  uint64_t xSize;
252  READDATA(is, xSize);
253  nSizeRet = xSize;
254  if (nSizeRet < 0x100000000ULL)
255  throw std::ios_base::failure("non-canonical ReadCompactSize()");
256  }
257  if (nSizeRet > (uint64_t)MAX_SIZE)
258  throw std::ios_base::failure("ReadCompactSize() : size too large");
259  return nSizeRet;
260 }

262 // Variable-length integers: bytes are a MSB base-128 encoding of the number.
263 // The high bit in each byte signifies whether another digit follows. To make
264 // the encoding is one-to-one, one is subtracted from all but the last digit.
265 // Thus, the byte sequence a[] with length len, where all but the last byte
266 // has bit 128 set, encodes the number:
267 //
268 // (a[len-1] & 0x7F) + sum(i=1..len-1, 128^i*((a[len-i-1] & 0x7F)+1))
269 //
270 // Properties:
271 // * Very small (0-127: 1 byte, 128-16511: 2 bytes, 16512-2113663: 3 bytes)
272 // * Every integer has exactly one encoding
273 // * Encoding does not depend on size of original integer type
274 // * No redundancy: every (infinite) byte sequence corresponds to a list
275 // of encoded integers.
276 //
277 // 0: [0x00] 256: [0x81 0x00]
278 // 1: [0x01] 16383: [0xFE 0x7F]
279 // 127: [0x7F] 16384: [0xFF 0x00]
280 // 128: [0x80 0x00] 16511: [0x80 0xFF 0x7F]
281 // 255: [0x80 0x7F] 65535: [0x82 0xFD 0x7F]
282 // 2^32: [0x8E 0xFE 0xFE 0xFF 0x00]
283 
284 template<typename I>
285 inline unsigned int GetSizeOfVarInt(I n)
286 {
287  int nRet = 0;
288  while(true) {
289  nRet++;
290  if (n <= 0x7F)
291  break;
292  n = (n >> 7) - 1;
293  }
294  return nRet;
295 }

297 template<typename Stream, typename I>
298 void WriteVarInt(Stream& os, I n)
299 {
300  unsigned char tmp[(sizeof(n)*8+6)/7];
301  int len=0;
302  while(true) {
303  tmp[len] = (n & 0x7F) | (len ? 0x80 : 0x00);
304  if (n <= 0x7F)
305  break;
306  n = (n >> 7) - 1;
307  len++;
308  }
309  do {
310  WRITEDATA(os, tmp[len]);
311  } while(len--);
312 }

314 template<typename Stream, typename I>
315 I ReadVarInt(Stream& is)
316 {
317  I n = 0;
318  while(true) {
319  unsigned char chData;
320  READDATA(is, chData);
321  n = (n << 7) | (chData & 0x7F);
322  if (chData & 0x80)
323  n++;
324  else
325  return n;
326  }
