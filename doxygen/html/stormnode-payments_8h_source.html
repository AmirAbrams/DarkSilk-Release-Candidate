115  bool GetPayee(CScript& payee)
116  {
117  LOCK(cs_vecPayments);
118 
119  int nVotes = -1;
120  BOOST_FOREACH(CStormnodePayee& p, vecPayments){
121  if(p.nVotes > nVotes){

124  }
125  }
126 
127  return (nVotes > -1);
128  }
