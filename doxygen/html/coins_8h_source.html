185  unsigned int GetSerializeSize(int nType, int nVersion) const {
186  unsigned int nSize = 0;
187  unsigned int nMaskSize = 0, nMaskCode = 0;
188  CalcMaskSize(nMaskSize, nMaskCode);
189  bool fFirst = vout.size() > 0 && !vout[0].IsNull();
190  bool fSecond = vout.size() > 1 && !vout[1].IsNull();
191  assert(fFirst || fSecond || nMaskCode);
192  unsigned int nCode = 8*(nMaskCode - (fFirst || fSecond ? 0 : 1)) + (fCoinBase ? 1 : 0) + (fFirst ? 2 : 0) + (fSecond ? 4 : 0);
193  // version
194  nSize += ::GetSerializeSize(VARINT(this->nVersion), nType, nVersion);
195  // size of header code
196  nSize += ::GetSerializeSize(VARINT(nCode), nType, nVersion);
197  // spentness bitmask
198  nSize += nMaskSize;
199  // txouts themself

201  for (unsigned int i = 0; i < vout.size(); i++)
202  if (!vout[i].IsNull())

210  template<typename Stream>
211  void Serialize(Stream &s, int nType, int nVersion) const {
212  unsigned int nMaskSize = 0, nMaskCode = 0;
213  CalcMaskSize(nMaskSize, nMaskCode);
214  bool fFirst = vout.size() > 0 && !vout[0].IsNull();
215  bool fSecond = vout.size() > 1 && !vout[1].IsNull();
216  assert(fFirst || fSecond || nMaskCode);
217  unsigned int nCode = 8*(nMaskCode - (fFirst || fSecond ? 0 : 1)) + (fCoinBase ? 1 : 0) + (fFirst ? 2 : 0) + (fSecond ? 4 : 0);
218  // version
219  ::Serialize(s, VARINT(this->nVersion), nType, nVersion);
220  // header code
221  ::Serialize(s, VARINT(nCode), nType, nVersion);
222  // spentness bitmask
223  for (unsigned int b = 0; b<nMaskSize; b++) {
224  unsigned char chAvail = 0;
225  for (unsigned int i = 0; i < 8 && 2+b*8+i < vout.size(); i++)
226  if (!vout[2+b*8+i].IsNull())
227  chAvail |= (1 << i);
228  ::Serialize(s, chAvail, nType, nVersion);
229  }
230  // txouts themself
231  for (unsigned int i = 0; i < vout.size(); i++) {
232  if (!vout[i].IsNull())
233  ::Serialize(s, CTxOutCompressorCoin(REF(vout[i])), nType, nVersion);
234  }
235  // coinbase height
236  ::Serialize(s, VARINT(nHeight), nType, nVersion);
237  }

239  template<typename Stream>
240  void Unserialize(Stream &s, int nType, int nVersion) {
241  unsigned int nCode = 0;
242  // version
243  ::Unserialize(s, VARINT(this->nVersion), nType, nVersion);
244  // header code
245  ::Unserialize(s, VARINT(nCode), nType, nVersion);
246  fCoinBase = nCode & 1;
247  std::vector<bool> vAvail(2, false);
248  vAvail[0] = (nCode & 2) != 0;
249  vAvail[1] = (nCode & 4) != 0;
250  unsigned int nMaskCode = (nCode / 8) + ((nCode & 6) != 0 ? 0 : 1);
251  // spentness bitmask
252  while (nMaskCode > 0) {
253  unsigned char chAvail = 0;
254  ::Unserialize(s, chAvail, nType, nVersion);
255  for (unsigned int p = 0; p < 8; p++) {
256  bool f = (chAvail & (1 << p)) != 0;
257  vAvail.push_back(f);
258  }
259  if (chAvail != 0)
260  nMaskCode--;
261  }
262  // txouts themself
263  vout.assign(vAvail.size(), CTxOut());
264  for (unsigned int i = 0; i < vAvail.size(); i++) {
265  if (vAvail[i])
266  ::Unserialize(s, REF(CTxOutCompressorCoin(vout[i])), nType, nVersion);
267  }
268  // coinbase height
269  ::Unserialize(s, VARINT(nHeight), nType, nVersion);
270  Cleanup();
271  }
