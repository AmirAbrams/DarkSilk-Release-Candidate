173  void seenTxConfirm(const CFeeRate& feeRate, const CFeeRate& minRelayFee, double dPriority, int nBlocksAgo)
174  {
175  // Last entry records "everything else".
176  int nBlocksTruncated = min(nBlocksAgo, (int) history.size() - 1);
177  assert(nBlocksTruncated >= 0);
178 
179  // We need to guess why the transaction was included in a block-- either
180  // because it is high-priority or because it has sufficient fees.
181  bool sufficientFee = (feeRate > minRelayFee);
182  bool sufficientPriority = AllowFree(dPriority);
183  const char* assignedTo = "unassigned";
184  if (sufficientFee && !sufficientPriority && CBlockAverage::AreSane(feeRate, minRelayFee))
185  {
186  history[nBlocksTruncated].RecordFee(feeRate);
187  assignedTo = "fee";
188  }
189  else if (sufficientPriority && !sufficientFee && CBlockAverage::AreSane(dPriority))
190  {
191  history[nBlocksTruncated].RecordPriority(dPriority);
192  assignedTo = "priority";
193  }
194  else
195  {
196  // Neither or both fee and priority sufficient to get confirmed:
197  // don't know why they got confirmed.
198  }
199  LogPrint("estimatefee", "Seen TX confirm: %s : %s fee/%g priority, took %d blocks\n",
200  assignedTo, feeRate.ToString(), dPriority, nBlocksAgo);
201  }

209  void seenBlock(const std::vector<CTxMemPoolEntry>& entries, int nBlockHeight, const CFeeRate minRelayFee)
210  {
211  if (nBlockHeight <= nBestSeenHeight)
212  {
213  // Ignore side chains and re-orgs; assuming they are random
214  // they don't affect the estimate.
215  // And if an attacker can re-org the chain at will, then
216  // you've got much bigger problems than "attacker can influence
217  // transaction fees."
218  return;
219  }
220  nBestSeenHeight = nBlockHeight;
221 
222  // Fill up the history buckets based on how long transactions took
223  // to confirm.
224  std::vector<std::vector<const CTxMemPoolEntry*> > entriesByConfirmations;
225  entriesByConfirmations.resize(history.size());
226  BOOST_FOREACH(const CTxMemPoolEntry& entry, entries)
227  {
228  // How many blocks did it take for miners to include this transaction?
229  int delta = nBlockHeight - entry.GetHeight();
230  if (delta <= 0)
231  {
232  // Re-org made us lose height, this should only happen if we happen
233  // to re-org on a difficulty transition point: very rare!
234  continue;
235  }
236  if ((delta-1) >= (int)history.size())
237  delta = history.size(); // Last bucket is catch-all
238  entriesByConfirmations.at(delta-1).push_back(&entry);
239  }
240  for (size_t i = 0; i < entriesByConfirmations.size(); i++)
241  {
242  std::vector<const CTxMemPoolEntry*> &e = entriesByConfirmations.at(i);
243  // Insert at most 10 random entries per bucket, otherwise a single block
244  // can dominate an estimate:
245  if (e.size() > 10) {
246  std::random_shuffle(e.begin(), e.end());
247  e.resize(10);
248  }
249  BOOST_FOREACH(const CTxMemPoolEntry* entry, e)
250  {
251  // Fees are stored and reported as BTC-per-kb:
252  CFeeRate feeRate(entry->GetFee(), entry->GetTxSize());
253  double dPriority = entry->GetPriority(entry->GetHeight()); // Want priority when it went IN
254  seenTxConfirm(feeRate, minRelayFee, dPriority, i);
255  }
256  }
257 
258  //After new samples are added, we have to clear the sorted lists,
259  //so they'll be resorted the next time someone asks for an estimate
260  sortedFeeSamples.clear();
261  sortedPrioritySamples.clear();
262 
263  for (size_t i = 0; i < history.size(); i++) {
264  if (history[i].FeeSamples() + history[i].PrioritySamples() > 0)
265  LogPrint("estimatefee", "estimates: for confirming within %d blocks based on %d/%d samples, fee=%s, prio=%g\n",
266  i,
267  history[i].FeeSamples(), history[i].PrioritySamples(),
268  estimateFee(i+1).ToString(), estimatePriority(i+1));
269  }
270  }

275  CFeeRate estimateFee(int nBlocksToConfirm)
276  {
277  nBlocksToConfirm--;
278 
279  if (nBlocksToConfirm < 0 || nBlocksToConfirm >= (int)history.size())
280  return CFeeRate(0);
281 
282  if (sortedFeeSamples.size() == 0)
283  {
284  for (size_t i = 0; i < history.size(); i++)
285  history.at(i).GetFeeSamples(sortedFeeSamples);
286  std::sort(sortedFeeSamples.begin(), sortedFeeSamples.end(),
287  std::greater<CFeeRate>());
288  }
289  if (sortedFeeSamples.size() < 11)
290  {
291  // Eleven is Gavin's Favorite Number
292  // ... but we also take a maximum of 10 samples per block so eleven means
293  // we're getting samples from at least two different blocks
294  return CFeeRate(0);
295  }
296 
297  int nBucketSize = history.at(nBlocksToConfirm).FeeSamples();
298 
299  // Estimates should not increase as number of confirmations goes up,
300  // but the estimates are noisy because confirmations happen discretely
301  // in blocks. To smooth out the estimates, use all samples in the history
302  // and use the nth highest where n is (number of samples in previous bucket +
303  // half the samples in nBlocksToConfirm bucket):
304  size_t nPrevSize = 0;
305  for (int i = 0; i < nBlocksToConfirm; i++)
306  nPrevSize += history.at(i).FeeSamples();
307  size_t index = min(nPrevSize + nBucketSize/2, sortedFeeSamples.size()-1);
308  return sortedFeeSamples[index];
309  }
310  double estimatePriority(int nBlocksToConfirm)
311  {
312  nBlocksToConfirm--;
313 
314  if (nBlocksToConfirm < 0 || nBlocksToConfirm >= (int)history.size())
315  return -1;
316 
317  if (sortedPrioritySamples.size() == 0)
318  {
319  for (size_t i = 0; i < history.size(); i++)
320  history.at(i).GetPrioritySamples(sortedPrioritySamples);
321  std::sort(sortedPrioritySamples.begin(), sortedPrioritySamples.end(),
322  std::greater<double>());
323  }
324  if (sortedPrioritySamples.size() < 11)
325  return -1.0;
326 
327  int nBucketSize = history.at(nBlocksToConfirm).PrioritySamples();
328 
329  // Estimates should not increase as number of confirmations needed goes up,
330  // but the estimates are noisy because confirmations happen discretely
331  // in blocks. To smooth out the estimates, use all samples in the history
332  // and use the nth highest where n is (number of samples in previous buckets +
333  // half the samples in nBlocksToConfirm bucket).
334  size_t nPrevSize = 0;
335  for (int i = 0; i < nBlocksToConfirm; i++)
336  nPrevSize += history.at(i).PrioritySamples();
337  size_t index = min(nPrevSize + nBucketSize/2, sortedPrioritySamples.size()-1);
338  return sortedPrioritySamples[index];
339  }
