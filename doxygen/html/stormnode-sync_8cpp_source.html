173 void CStormnodeSync::ProcessMessage(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
174 {
175  if (strCommand == "ssc") { //Sync status count
176  int nItemID;
177  int nCount;
178  vRecv >> nItemID >> nCount;
179 
180  if(RequestedStormnodeAssets >= STORMNODE_SYNC_FINISHED) return;
181 
182  //this means we will receive no further communication
183  switch(nItemID)
184  {
185  case(STORMNODE_SYNC_LIST):
186  if(nItemID != RequestedStormnodeAssets) return;
187  sumStormnodeList += nCount;
188  countStormnodeList++;
189  break;
190  case(STORMNODE_SYNC_SNW):
191  if(nItemID != RequestedStormnodeAssets) return;
192  sumStormnodeWinner += nCount;
193  countStormnodeWinner++;
194  break;
195  case(STORMNODE_SYNC_BUDGET_PROP):
196  if(RequestedStormnodeAssets != STORMNODE_SYNC_BUDGET) return;
197  sumBudgetItemProp += nCount;
198  countBudgetItemProp++;
199  break;
200  case(STORMNODE_SYNC_BUDGET_FIN):
201  if(RequestedStormnodeAssets != STORMNODE_SYNC_BUDGET) return;
202  sumBudgetItemFin += nCount;
203  countBudgetItemFin++;
204  break;
205  }
206 
207  LogPrintf("CStormnodeSync:ProcessMessage - ssc - got inventory count %d %d\n", nItemID, nCount);
208  }
