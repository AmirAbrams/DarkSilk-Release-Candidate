220  bool IsDust(CFeeRate minRelayTxFee) const
221  {
222  // "Dust" is defined in terms of CTransaction::minRelayTxFee, which has units duffs-per-kilobyte.
223  // If you'd pay more than 1/3 in fees to spend something, then we consider it dust.
224  // A typical txout is 34 bytes big, and will need a CTxIn of at least 148 bytes to spend
225  // i.e. total is 148 + 32 = 182 bytes. Default -minrelaytxfee is 10000 duffs per kB
226  // and that means that fee per txout is 182 * 10000 / 1000 = 1820 duffs.
227  // So dust is a txout less than 1820 *3 = 5460 duffs
228  // with default -minrelaytxfee = minRelayTxFee = 10000 duffs per kB.
229  size_t nSize = GetSerializeSize(SER_DISK,0)+148u;
230  return (nValue < 3*minRelayTxFee.GetFee(nSize));
231  }
