592

594

596  {

598

600  {
601  //if (fDebugSmsg)

603  if (it->first < cutoffTime)

607 
608  std::string fileName = boost::lexical_cast<std::string>(it->first);
609 
610  fs::path fullPath = GetDataDir() / "smsgStore" / (fileName + "_01.dat");
611  if (fs::exists(fullPath))
612  {
613  try { fs::remove(fullPath);
614  } catch (const fs::filesystem_error& ex)
615  {
616  LogPrintf("Error removing bucket file %s.\n", ex.what());
617  };
618  } else
619  {
620  LogPrintf("Path %s does not exist \n", fullPath.string().c_str());
621  };
622 
623  // -- look for a wl file, it stores incoming messages when wallet is locked
624  fullPath = GetDataDir() / "smsgStore" / (fileName + "_01_wl.dat");
625  if (fs::exists(fullPath))
626  {
627  try { fs::remove(fullPath);
628  } catch (const fs::filesystem_error& ex)

643 
644  it->second.nLockPeerId = 0;
645  }; // if (it->second.nLockCount == 0)
646 
647  }; // ! if (it->first < cutoffTime)
648  };
649  } // cs_smsg

1410  if (smsgBuckets[time].setTokens.size() < ncontent
1411  || (smsgBuckets[time].setTokens.size() == ncontent
1412  && smsgBuckets[time].hash != hash)) // if same amount in buckets check hash

1421  nShowBuckets++;
1422  };

1424  };
