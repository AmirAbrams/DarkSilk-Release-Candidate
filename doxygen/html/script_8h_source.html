397  explicit CScriptNum(const std::vector<unsigned char>& vch, bool fRequireMinimal)
398  {
399  if (vch.size() > nMaxNumSize) {
400  throw scriptnum_error("script number overflow");
401  }
402  if (fRequireMinimal && vch.size() > 0) {
403  // Check that the number is encoded with the minimum possible
404  // number of bytes.
405  //
406  // If the most-significant-byte - excluding the sign bit - is zero
407  // then we're not minimal. Note how this test also rejects the
408  // negative-zero encoding, 0x80.
409  if ((vch.back() & 0x7f) == 0) {
410  // One exception: if there's more than one byte and the most
411  // significant bit of the second-most-significant-byte is set
412  // it would conflict with the sign bit. An example of this case
413  // is +-255, which encode to 0xff00 and 0xff80 respectively.
414  // (big-endian).
415  if (vch.size() <= 1 || (vch[vch.size() - 2] & 0x80) == 0) {
416  throw scriptnum_error("non-minimally encoded script number");
417  }
418  }
419  }
420  m_value = set_vch(vch);
421  }

487  static std::vector<unsigned char> serialize(const int64_t& value)
488  {
489  if(value == 0)
490  return std::vector<unsigned char>();
491 
492  std::vector<unsigned char> result;
493  const bool neg = value < 0;
494  uint64_t absvalue = neg ? -value : value;
495 
496  while(absvalue)
497  {
498  result.push_back(absvalue & 0xff);
499  absvalue >>= 8;
500  }
501 
502 // - If the most significant byte is >= 0x80 and the value is positive, push a
503 // new zero-byte to make the significant byte < 0x80 again.
504 
505 // - If the most significant byte is >= 0x80 and the value is negative, push a
506 // new 0x80 byte that will be popped off when converting to an integral.
507 
508 // - If the most significant byte is < 0x80 and the value is negative, add
509 // 0x80 to it, since it will be subtracted and interpreted as a negative when
510 // converting to an integral.
511 
512  if (result.back() & 0x80)
513  result.push_back(neg ? 0x80 : 0);
514  else if (neg)
515  result.back() |= 0x80;
516 
517  return result;
518  }

522 private:
523  static int64_t set_vch(const std::vector<unsigned char>& vch)
524  {
525  if (vch.empty())
526  return 0;
527 
528  int64_t result = 0;
529  for (size_t i = 0; i != vch.size(); ++i)
530  result |= static_cast<int64_t>(vch[i]) << 8*i;
531 
532  // If the input vector's most significant byte is 0x80, remove it from
533  // the result's msb and return a negative.
534  if (vch.back() & 0x80)
535  return -((int64_t)(result & ~(0x80ULL << (8 * (vch.size() - 1)))));
536 
537  return result;
538  }

729  bool GetOp2(const_iterator& pc, opcodetype& opcodeRet, std::vector<unsigned char>* pvchRet) const
730  {
731  opcodeRet = OP_INVALIDOPCODE;
732  if (pvchRet)
733  pvchRet->clear();
734  if (pc >= end())
735  return false;
736 
737  // Read instruction
738  if (end() - pc < 1)
739  return false;
740  unsigned int opcode = *pc++;
741 
742  // Immediate operand
743  if (opcode <= OP_PUSHDATA4)
744  {
745  unsigned int nSize;
746  if (opcode < OP_PUSHDATA1)
747  {
748  nSize = opcode;
749  }
750  else if (opcode == OP_PUSHDATA1)
751  {
752  if (end() - pc < 1)
753  return false;
754  nSize = *pc++;
755  }
756  else if (opcode == OP_PUSHDATA2)
757  {
758  if (end() - pc < 2)
759  return false;
760  nSize = 0;
761  memcpy(&nSize, &pc[0], 2);
762  pc += 2;
763  }
764  else if (opcode == OP_PUSHDATA4)
765  {
766  if (end() - pc < 4)
767  return false;
768  memcpy(&nSize, &pc[0], 4);
769  pc += 4;
770  }
771  if (end() - pc < 0 || (unsigned int)(end() - pc) < nSize)
772  return false;
773  if (pvchRet)
774  pvchRet->assign(pc, pc + nSize);
775  pc += nSize;
776  }
777 
778  opcodeRet = (opcodetype)opcode;
779  return true;
780  }
