48 // This stores payment requests received before

54 //
55 // Sending to the server is done synchronously, at startup.
56 // If the server isn't already running, startup continues,
57 // and the items in savedPaymentRequest will be handled
58 // when uiReady() is called.
59 //
60 bool PaymentServer::ipcSendCommandLine()
61 {
62  bool fResult = false;
63 
64  const QStringList& args = qApp->arguments();
65  for (int i = 1; i < args.size(); i++)
66  {
67  if (!args[i].startsWith(DARKSILK_IPC_PREFIX, Qt::CaseInsensitive))
68  continue;
69  savedPaymentRequests.append(args[i]);
70  }
71 
72  foreach (const QString& arg, savedPaymentRequests)
73  {
74  QLocalSocket* socket = new QLocalSocket();
75  socket->connectToServer(ipcServerName(), QIODevice::WriteOnly);
76  if (!socket->waitForConnected(DARKSILK_IPC_CONNECT_TIMEOUT))
77  return false;
78 
79  QByteArray block;
80  QDataStream out(&block, QIODevice::WriteOnly);
81  out.setVersion(QDataStream::Qt_4_0);
82  out << arg;
83  out.device()->seek(0);
84  socket->write(block);
85  socket->flush();
86 
87  socket->waitForBytesWritten(DARKSILK_IPC_CONNECT_TIMEOUT);
88  socket->disconnectFromServer();
89  delete socket;
90  fResult = true;
91  }
92  return fResult;
93 }
