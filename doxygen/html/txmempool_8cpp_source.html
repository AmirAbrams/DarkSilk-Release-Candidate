45 bool CTxMemPool::addUnchecked(const uint256& hash, CTransaction &tx)
46 {
47  // Add to memory pool without checking anything.
48  // Used by main.cpp AcceptToMemoryPool(), which DOES do
49  // all the appropriate checks.
50  LOCK(cs);
51  {
52  mapTx[hash] = tx;
53  for (unsigned int i = 0; i < tx.vin.size(); i++)
54  mapNextTx[tx.vin[i].prevout] = CInPoint(&mapTx[hash], i);
55  nTransactionsUpdated++;
56  }
57  return true;
58 }

60 bool CTxMemPool::remove(const CTransaction &tx, bool fRecursive)
61 {
62  // Remove transaction from memory pool
63  {
64  LOCK(cs);
65  uint256 hash = tx.GetHash();
66  if (mapTx.count(hash))
67  {
68  if (fRecursive) {
69  for (unsigned int i = 0; i < tx.vout.size(); i++) {
70  std::map<COutPoint, CInPoint>::iterator it = mapNextTx.find(COutPoint(hash, i));
71  if (it != mapNextTx.end())
72  remove(*it->second.ptx, true);
73  }
74  }
75  BOOST_FOREACH(const CTxIn& txin, tx.vin)
76  mapNextTx.erase(txin.prevout);
77  mapTx.erase(hash);
78  nTransactionsUpdated++;
79  }
80  }
81  return true;
82 }

84 bool CTxMemPool::removeConflicts(const CTransaction &tx)
85 {
86  // Remove transactions which depend on inputs of tx, recursively
87  LOCK(cs);
88  BOOST_FOREACH(const CTxIn &txin, tx.vin) {
89  std::map<COutPoint, CInPoint>::iterator it = mapNextTx.find(txin.prevout);
90  if (it != mapNextTx.end()) {
91  const CTransaction &txConflict = *it->second.ptx;
92  if (txConflict != tx)
93  remove(txConflict, true);
94  }
95  }
96  return true;
97 }
