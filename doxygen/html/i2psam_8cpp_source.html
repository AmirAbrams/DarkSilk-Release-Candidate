292 std::string StreamSession::generateSessionID()
293 {
294  static const int minSessionIDLength = 5;
295  static const int maxSessionIDLength = 9;
296  static const char sessionIDAlphabet[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
297  int length = minSessionIDLength - 1;
298  std::string result;
299 
300  srand(time(NULL));
301 
302  while(length < minSessionIDLength)
