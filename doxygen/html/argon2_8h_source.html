145 /*
146  *****Context: structure to hold Argon2 inputs:
147  * output array and its length,
148  * password and its length,
149  * salt and its length,
150  * secret and its length,
151  * associated data and its length,
152  * number of passes, amount of used memory (in KBytes, can be rounded up a bit)
153  * number of parallel threads that will be run.
154  * All the parameters above affect the output hash value.
155  * Additionally, two function pointers can be provided to allocate and
156  deallocate the memory (if NULL, memory will be allocated internally).
157  * Also, three flags indicate whether to erase password, secret as soon as they
158  are pre-hashed (and thus not needed anymore), and the entire memory
159  ****************************
160  Simplest situation: you have output array out[8], password is stored in
161  pwd[32], salt is stored in salt[16], you do not have keys nor associated data.
162  You need to spend 1 GB of RAM and you run 5 passes of Argon2d with 4 parallel
163  lanes.
164  You want to erase the password, but you're OK with last pass not being erased.
165  You want to use the default memory allocator.
166  Then you initialize
167  Argon2_Context(out,8,pwd,32,salt,16,NULL,0,NULL,0,5,1<<20,4,4,NULL,NULL,true,false,false,false).
168  */
169 typedef struct Argon2_Context {
170  uint8_t *out; /* output array */
171  uint32_t outlen; /* digest length */
172 
173  uint8_t *pwd; /* password array */
174  uint32_t pwdlen; /* password length */
175 
176  uint8_t *salt; /* salt array */
177  uint32_t saltlen; /* salt length */
178 
179  uint8_t *secret; /* key array */
180  uint32_t secretlen; /* key length */
181 
182  uint8_t *ad; /* associated data array */
183  uint32_t adlen; /* associated data length */
184 
185  uint32_t t_cost; /* number of passes */
186  uint32_t m_cost; /* amount of memory requested (KB) */
187  uint32_t lanes; /* number of lanes */
188  uint32_t threads; /* maximum number of threads */
189 
190  allocate_fptr allocate_cbk; /* pointer to memory allocator */
191  deallocate_fptr free_cbk; /* pointer to memory deallocator */
192 
193  uint32_t flags; /* array of bool options */
194 } argon2_context;
