145 string CRPCTable::help(string strCommand) const
146 {
147  string strRet;
148  set<rpcfn_type> setDone;
149  for (map<string, const CRPCCommand*>::const_iterator mi = mapCommands.begin(); mi != mapCommands.end(); ++mi)
150  {
151  const CRPCCommand *pcmd = mi->second;
152  string strMethod = mi->first;
153  // We already filter duplicates, but these deprecated screw up the sort order
154  if (strMethod.find("label") != string::npos)
155  continue;
156  if (strCommand != "" && strMethod != strCommand)
157  continue;
158 #ifdef ENABLE_WALLET
159  if (pcmd->reqWallet && !pwalletMain)
160  continue;
161 #endif
162 
163  try
164  {
165  Array params;
166  rpcfn_type pfn = pcmd->actor;
167  if (setDone.insert(pfn).second)
168  (*pfn)(params, true);
169  }
170  catch (std::exception& e)
171  {
172  // Help text is returned in an exception
173  string strHelp = string(e.what());
174  if (strCommand == "")
175  if (strHelp.find('\n') != string::npos)
176  strHelp = strHelp.substr(0, strHelp.find('\n'));
177  strRet += strHelp + "\n";
178  }
179  }
180  if (strRet == "")
181  strRet = strprintf("help: unknown command: %s\n", strCommand);
182  strRet = strRet.substr(0,strRet.size()-1);
183  return strRet;
184 }

375 bool ClientAllowed(const boost::asio::ip::address& address)
376 {
377  // Make sure that IPv4-compatible and IPv4-mapped IPv6 addresses are treated as IPv4 addresses
378  if (address.is_v6()
379  && (address.to_v6().is_v4_compatible()
380  || address.to_v6().is_v4_mapped()))
381  return ClientAllowed(address.to_v6().to_v4());
382 
383  if (address == asio::ip::address_v4::loopback()
384  || address == asio::ip::address_v6::loopback()
385  || (address.is_v4()
386  // Check whether IPv4 addresses match 127.0.0.0/8 (loopback subnet)
387  && (address.to_v4().to_ulong() & 0xff000000) == 0x7f000000))
388  return true;
389 
390  const string strAddress = address.to_string();
391  const vector<string>& vAllow = mapMultiArgs["-rpcallowip"];
392  BOOST_FOREACH(string strAllow, vAllow)
393  if (WildcardMatch(strAddress, strAllow))
394  return true;
395  return false;

743 void ServiceConnection(AcceptedConnection *conn)
744 {
745  bool fRun = true;
746  while (fRun)
747  {
748  int nProto = 0;
749  map<string, string> mapHeaders;
750  string strRequest, strMethod, strURI;
751 
752  // Read HTTP request line
753  if (!ReadHTTPRequestLine(conn->stream(), nProto, strMethod, strURI))
754  break;
755 
756  // Read HTTP message headers and body
757  ReadHTTPMessage(conn->stream(), mapHeaders, strRequest, nProto, MAX_SIZE);
758 
759  if (strURI != "/") {
760  conn->stream() << HTTPReply(HTTP_NOT_FOUND, "", false) << std::flush;
761  break;
762  }
763 
764  // Check authorization
765  if (mapHeaders.count("authorization") == 0)
766  {
767  conn->stream() << HTTPReply(HTTP_UNAUTHORIZED, "", false) << std::flush;
768  break;
769  }
770  if (!HTTPAuthorized(mapHeaders))
771  {
772  LogPrintf("ThreadRPCServer incorrect password attempt from %s\n", conn->peer_address_to_string());
773  /* Deter brute-forcing short passwords.
774  If this results in a DoS the user really
775  shouldn't have their RPC port exposed. */
776  if (mapArgs["-rpcpassword"].size() < 20)
777  MilliSleep(250);
778 
779  conn->stream() << HTTPReply(HTTP_UNAUTHORIZED, "", false) << std::flush;
780  break;
781  }
782  if (mapHeaders["connection"] == "close")
783  fRun = false;
784 
785  JSONRequest jreq;
786  try
787  {
788  // Parse request
789  Value valRequest;
790  if (!read_string(strRequest, valRequest))
791  throw JSONRPCError(RPC_PARSE_ERROR, "Parse error");
792 
793  string strReply;
794 
795  // singleton request
796  if (valRequest.type() == obj_type) {
797  jreq.parse(valRequest);
798 
799  Value result = tableRPC.execute(jreq.strMethod, jreq.params);
800 
801  // Send reply
802  strReply = JSONRPCReply(result, Value::null, jreq.id);
803 
804  // array of requests
805  } else if (valRequest.type() == array_type)
806  strReply = JSONRPCExecBatch(valRequest.get_array());
807  else
808  throw JSONRPCError(RPC_PARSE_ERROR, "Top-level object parse error");
809 
810  conn->stream() << HTTPReply(HTTP_OK, strReply, fRun) << std::flush;
811  }
812  catch (Object& objError)
813  {
814  ErrorReply(conn->stream(), objError, jreq.id);
815  break;
816  }
817  catch (std::exception& e)
818  {
819  ErrorReply(conn->stream(), JSONRPCError(RPC_PARSE_ERROR, e.what()), jreq.id);
820  break;
821  }
822  }
823 }
