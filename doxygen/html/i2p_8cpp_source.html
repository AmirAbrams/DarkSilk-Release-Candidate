193 std::string I2PSession::GenerateB32AddressFromDestination(const std::string& destination)
194 {
195  std::string canonicalDest = destination;
196  for (size_t pos = canonicalDest.find_first_of('-'); pos != std::string::npos; pos = canonicalDest.find_first_of('-', pos))
197  canonicalDest[pos] = '+';
198  for (size_t pos = canonicalDest.find_first_of('~'); pos != std::string::npos; pos = canonicalDest.find_first_of('~', pos))
199  canonicalDest[pos] = '/';
200  std::string rawHash = DecodeBase64(canonicalDest);
201  uint256 hash;
202  SHA256((const unsigned char*)rawHash.c_str(), rawHash.size(), (unsigned char*)&hash);
203  std::string result = EncodeBase32(hash.begin(), hash.end() - hash.begin()) + ".b32.i2p";
204  for (size_t pos = result.find_first_of('='); pos != std::string::npos; pos = result.find_first_of('=', pos-1))
205  result.erase(pos, 1);
206  return result;
207 }
