346 std::string HexStr(const T itbegin, const T itend, bool fSpaces=false)
347 {
348  std::string rv;
349  static const char hexmap[16] = { '0', '1', '2', '3', '4', '5', '6', '7',
350  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f' };
351  rv.reserve((itend-itbegin)*3);
