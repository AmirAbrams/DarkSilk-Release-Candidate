860 void DarkSilkGUI::setNumBlocks(int count)
861 {
862  // don't show / hide progress bar and its label if we have no connection to the network
863  if (!clientModel || (clientModel->getNumConnections() == 0 && !clientModel->isImporting()))
864  {
865  statusBar()->setVisible(false);
866 
867  return;
868  }
869 
870  QString tooltip;
871 
872  QDateTime lastBlockDate = clientModel->getLastBlockDate();
873  QDateTime currentDate = QDateTime::currentDateTime();
874  int totalSecs = GetTime() - Params().GenesisBlock().GetBlockTime();
875  int secs = lastBlockDate.secsTo(currentDate);
876 
877  tooltip = tr("Processed %1 blocks of transaction history.").arg(count);
878 
879  // Set icon state: spinning if catching up, tick otherwise
880  if(secs < 25 * 60) {
881  tooltip = tr("Up to date") + QString(".<br>") + tooltip;
882 
883  labelBlocksIcon->setPixmap(QIcon(fUseBlackTheme ? ":/icons/black/synced" : ":/icons/synced").pixmap(STATUSBAR_ICONSIZE, STATUSBAR_ICONSIZE));
884  overviewPage->showOutOfSyncWarning(false);
885  progressBarLabel->setVisible(false);
886  progressBar->setVisible(false);
887  } else {
888  // Represent time from last generated block in human readable text
889  QString timeBehindText;
890  const int HOUR_IN_SECONDS = 60 * 60;
891  const int DAY_IN_SECONDS = 24 * 60 * 60;
892  const int WEEK_IN_SECONDS = 7 * 24 * 60 * 60;
893  const int YEAR_IN_SECONDS = 31556952; // Average length of year in Gregorian calendar
894 
895  if(secs < 2 * DAY_IN_SECONDS) {
896  timeBehindText = tr("%n hour(s)", "", secs / HOUR_IN_SECONDS);
897  } else if(secs < 2 * WEEK_IN_SECONDS) {
898  timeBehindText = tr("%n day(s)", "", secs / DAY_IN_SECONDS);
899  } else if(secs < YEAR_IN_SECONDS) {
900  timeBehindText = tr("%n week(s)", "", secs / WEEK_IN_SECONDS);
901  } else {
902  int years = secs / YEAR_IN_SECONDS;
903  int remainder = secs % YEAR_IN_SECONDS;
904 
905  timeBehindText = tr("%1 and %2").arg(tr("%n year(s)", "", years)).arg(tr("%n week(s)", "", remainder / WEEK_IN_SECONDS));
906  }
907 
908  progressBarLabel->setText(tr(clientModel->isImporting() ? "Importing blocks..." : "Synchronizing with network..."));
909  progressBarLabel->setVisible(true);
910 
911  progressBar->setFormat(tr("%1 behind").arg(timeBehindText));
912  progressBar->setMaximum(totalSecs);
913  progressBar->setValue(totalSecs - secs);
914  progressBar->setVisible(true);
915  //fShowStatusBar = true;
916 
917  tooltip = tr("Catching up...") + QString("<br>") + tooltip;
918 
919  labelBlocksIcon->setMovie(syncIconMovie);
920 
921  if(count != prevBlocks)
922  {
923  syncIconMovie->jumpToNextFrame();
924  }
925 
926  prevBlocks = count;
927 
928  overviewPage->showOutOfSyncWarning(true);
929 
930  tooltip += QString("<br>");
931  tooltip += tr("Last received block was generated %1 ago.").arg(timeBehindText);
932  tooltip += QString("<br>");
933  tooltip += tr("Transactions after this will not yet be visible.");
934  }
935 
936  // Don't word-wrap this (fixed-width) tooltip
937  tooltip = QString("<nobr>") + tooltip + QString("</nobr>");
938 
939  labelBlocksIcon->setToolTip(tooltip);
940  progressBarLabel->setToolTip(tooltip);
941  progressBar->setToolTip(tooltip);
942 
943  statusBar()->setVisible(true);
944 }
