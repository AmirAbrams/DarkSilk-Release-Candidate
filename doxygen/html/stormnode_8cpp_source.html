161  return 0;

164  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
165  ss << hash;
166  uint256 hash2 = ss.GetHash();
167 
168  CHashWriter ss2(SER_GETHASH, PROTOCOL_VERSION);
169  ss2 << hash;
