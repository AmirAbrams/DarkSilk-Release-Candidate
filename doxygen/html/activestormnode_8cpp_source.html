153 bool CActiveStormnode::SendStormnodePing(std::string& errorMessage) {
154  if(status != ACTIVE_STORMNODE_STARTED) {
155  errorMessage = "Stormnode is not in a running status";
156  return false;
157  }
158 
159  CPubKey pubKeyStormnode;
160  CKey keyStormnode;
161 
162  if(!sandStormSigner.SetKey(strStormNodePrivKey, errorMessage, keyStormnode, pubKeyStormnode))
163  {
164  errorMessage = strprintf("Error upon calling SetKey: %s\n", errorMessage);
165  return false;
166  }
167 
168  LogPrintf("CActiveStormnode::SendStormnodePing() - Relay Stormnode Ping vin = %s\n", vin.ToString());
169 
170  CStormnodePing snp(vin);
171  if(!snp.Sign(keyStormnode, pubKeyStormnode))
172  {
173  errorMessage = "Couldn't sign Stormnode Ping";
174  return false;
175  }
176 
177  // Update lastPing for our stormnode in Stormnode list
178  CStormnode* psn = snodeman.Find(vin);
179  if(psn != NULL)
180  {
181  if(psn->IsPingedWithin(STORMNODE_PING_SECONDS, snp.sigTime)){
182  errorMessage = "Too early to send Stormnode Ping";
183  return false;
184  }
185 
186  psn->lastPing = snp;
187  snodeman.mapSeenStormnodePing.insert(make_pair(snp.GetHash(), snp));
188 
189  //snodeman.mapSeenStormnodeBroadcast.lastPing is probably outdated, so we'll update it
190  CStormnodeBroadcast snb(*psn);
191  uint256 hash = snb.GetHash();
192  if(snodeman.mapSeenStormnodeBroadcast.count(hash)) snodeman.mapSeenStormnodeBroadcast[hash].lastPing = snp;
193 
194  snp.Relay();
195 
196  return true;
197  }
198  else
199  {
200  // Seems like we are trying to send a ping while the Stormnode is not registered in the network
201  errorMessage = "Sandstorm Stormnode List doesn't include our Stormnode, shutting down Stormnode pinging service! " + vin.ToString();
202  status = ACTIVE_STORMNODE_NOT_CAPABLE;
203  notCapableReason = errorMessage;
204  return false;
205  }

296 bool CActiveStormnode::GetStormNodeVin(CTxIn& vin, CPubKey& pubkey, CKey& secretKey, std::string strTxHash, std::string strOutputIndex) {
297  // Find possible candidates
298  TRY_LOCK(pwalletMain->cs_wallet, fWallet);
299  if(!fWallet) return false;
300 
301  vector<COutput> possibleCoins = SelectCoinsStormnode();
302  COutput *selectedOutput;
303 
304  // Find the vin
305  if(!strTxHash.empty()) {
306  // Let's find it
307  uint256 txHash(strTxHash);
308  int outputIndex = atoi(strOutputIndex.c_str());
309  bool found = false;
310  BOOST_FOREACH(COutput& out, possibleCoins) {
311  if(out.tx->GetHash() == txHash && out.i == outputIndex)
312  {
313  selectedOutput = &out;
314  found = true;
315  break;
316  }
317  }
318  if(!found) {
319  LogPrintf("CActiveStormnode::GetStormNodeVin - Could not locate valid vin\n");
320  return false;
321  }
322  } else {
323  // No output specified, Select the first one
324  if(possibleCoins.size() > 0) {
325  selectedOutput = &possibleCoins[0];
326  } else {
327  LogPrintf("CActiveStormnode::GetStormNodeVin - Could not locate specified vin from possible list\n");
328  return false;
329  }
330  }
331 
332  // At this point we have a selected output, retrieve the associated info
333  return GetVinFromOutput(*selectedOutput, vin, pubkey, secretKey);

365 vector<COutput> CActiveStormnode::SelectCoinsStormnode()
366 {
367  vector<COutput> vCoins;
368  vector<COutput> filteredCoins;
369  vector<COutPoint> confLockedCoins;
370 
371  // Temporary unlock SN coins from stormnode.conf

380  }
381 
382  // Retrieve all possible outputs
383  pwalletMain->AvailableCoins(vCoins);
384 
385  // Lock SN coins from stormnode.conf back if they where temporary unlocked
386  if(!confLockedCoins.empty()) {
387  BOOST_FOREACH(COutPoint outpoint, confLockedCoins)
388  pwalletMain->LockCoin(outpoint);
389  }
390 
391  // Filter
392  BOOST_FOREACH(const COutput& out, vCoins)
393  {
394  if(out.tx->vout[out.i].nValue == 10000*COIN) { //exactly

396  }
397  }
398  return filteredCoins;
