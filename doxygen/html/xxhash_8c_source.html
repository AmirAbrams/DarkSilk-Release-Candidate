34 //**************************************
35 // Tuning parameters
36 //**************************************
37 // Unaligned memory access is automatically enabled for "common" CPU, such as x86.
38 // For others CPU, the compiler will be more cautious, and insert extra code to ensure aligned access is respected.
39 // If you know your target CPU supports unaligned memory access, you want to force this option manually to improve performance.
40 // You can also enable this parameter if you know your input data will always be aligned (boundaries of 4, for U32).
41 #if defined(__ARM_FEATURE_UNALIGNED) || defined(__i386) || defined(_M_IX86) || defined(__x86_64__) || defined(_M_X64)
42 # define XXH_USE_UNALIGNED_ACCESS 1
43 #endif
44 
45 // XXH_ACCEPT_NULL_INPUT_POINTER :
46 // If the input pointer is a null pointer, xxHash default behavior is to trigger a memory access error, since it is a bad pointer.
47 // When this option is enabled, xxHash output for null input pointers will be the same as a null-length input.
48 // This option has a very small performance cost (only measurable on small inputs).
49 // By default, this option is disabled. To enable it, uncomment below define :
50 //#define XXH_ACCEPT_NULL_INPUT_POINTER 1
51 
52 // XXH_FORCE_NATIVE_FORMAT :
53 // By default, xxHash library provides endian-independant Hash values, based on little-endian convention.
54 // Results are therefore identical for little-endian and big-endian CPU.
55 // This comes at a performance cost for big-endian CPU, since some swapping is required to emulate little-endian format.
56 // Should endian-independance be of no importance for your application, you may set the #define below to 1.
57 // It will improve speed for Big-endian CPU.
58 // This option has no impact on Little_Endian CPU.
59 #define XXH_FORCE_NATIVE_FORMAT 0

207 FORCE_INLINE U32 XXH32_endian_align(const void* input, int len, U32 seed, XXH_endianess endian, XXH_alignment align)
208 {
209  const BYTE* p = (const BYTE*)input;
210  const BYTE* const bEnd = p + len;
211  U32 h32;
212 
213 #ifdef XXH_ACCEPT_NULL_INPUT_POINTER
214  if (p==NULL) { len=0; p=(const BYTE*)(size_t)16; }
215 #endif
216 
217  if (len>=16)
218  {
219  const BYTE* const limit = bEnd - 16;
220  U32 v1 = seed + PRIME32_1 + PRIME32_2;
221  U32 v2 = seed + PRIME32_2;
222  U32 v3 = seed + 0;
223  U32 v4 = seed - PRIME32_1;
224 
225  do
226  {
227  v1 += XXH_readLE32_align((const U32*)p, endian, align) * PRIME32_2; v1 = XXH_rotl32(v1, 13); v1 *= PRIME32_1; p+=4;
228  v2 += XXH_readLE32_align((const U32*)p, endian, align) * PRIME32_2; v2 = XXH_rotl32(v2, 13); v2 *= PRIME32_1; p+=4;
229  v3 += XXH_readLE32_align((const U32*)p, endian, align) * PRIME32_2; v3 = XXH_rotl32(v3, 13); v3 *= PRIME32_1; p+=4;
230  v4 += XXH_readLE32_align((const U32*)p, endian, align) * PRIME32_2; v4 = XXH_rotl32(v4, 13); v4 *= PRIME32_1; p+=4;
231  } while (p<=limit);
232 
233  h32 = XXH_rotl32(v1, 1) + XXH_rotl32(v2, 7) + XXH_rotl32(v3, 12) + XXH_rotl32(v4, 18);
234  }
235  else
236  {
237  h32 = seed + PRIME32_5;
238  }
239 
240  h32 += (U32) len;
241 
242  while (p<=bEnd-4)
243  {
244  h32 += XXH_readLE32_align((const U32*)p, endian, align) * PRIME32_3;
245  h32 = XXH_rotl32(h32, 17) * PRIME32_4 ;
246  p+=4;
247  }
248 
249  while (p<bEnd)
250  {
251  h32 += (*p) * PRIME32_5;
252  h32 = XXH_rotl32(h32, 11) * PRIME32_1 ;
253  p++;
254  }
255 
256  h32 ^= h32 >> 15;
257  h32 *= PRIME32_2;
258  h32 ^= h32 >> 13;
259  h32 *= PRIME32_3;
260  h32 ^= h32 >> 16;
261 
262  return h32;
263 }

340 FORCE_INLINE XXH_errorcode XXH32_update_endian (void* state_in, const void* input, int len, XXH_endianess endian)
341 {
342  struct XXH_state32_t * state = (struct XXH_state32_t *) state_in;
343  const BYTE* p = (const BYTE*)input;
344  const BYTE* const bEnd = p + len;
345 
346 #ifdef XXH_ACCEPT_NULL_INPUT_POINTER
347  if (input==NULL) return XXH_ERROR;
348 #endif
349 
350  state->total_len += len;
351 
352  if (state->memsize + len < 16) // fill in tmp buffer
353  {
354  XXH_memcpy(state->memory + state->memsize, input, len);
355  state->memsize += len;
356  return XXH_OK;
357  }
358 
359  if (state->memsize) // some data left from previous update
360  {
361  XXH_memcpy(state->memory + state->memsize, input, 16-state->memsize);
362  {
363  const U32* p32 = (const U32*)state->memory;
364  state->v1 += XXH_readLE32(p32, endian) * PRIME32_2; state->v1 = XXH_rotl32(state->v1, 13); state->v1 *= PRIME32_1; p32++;
365  state->v2 += XXH_readLE32(p32, endian) * PRIME32_2; state->v2 = XXH_rotl32(state->v2, 13); state->v2 *= PRIME32_1; p32++;
366  state->v3 += XXH_readLE32(p32, endian) * PRIME32_2; state->v3 = XXH_rotl32(state->v3, 13); state->v3 *= PRIME32_1; p32++;
367  state->v4 += XXH_readLE32(p32, endian) * PRIME32_2; state->v4 = XXH_rotl32(state->v4, 13); state->v4 *= PRIME32_1; p32++;
368  }
369  p += 16-state->memsize;
370  state->memsize = 0;
371  }
372 
373  if (p <= bEnd-16)
374  {
375  const BYTE* const limit = bEnd - 16;
376  U32 v1 = state->v1;
377  U32 v2 = state->v2;
378  U32 v3 = state->v3;
379  U32 v4 = state->v4;
380 
381  do
382  {
383  v1 += XXH_readLE32((const U32*)p, endian) * PRIME32_2; v1 = XXH_rotl32(v1, 13); v1 *= PRIME32_1; p+=4;
384  v2 += XXH_readLE32((const U32*)p, endian) * PRIME32_2; v2 = XXH_rotl32(v2, 13); v2 *= PRIME32_1; p+=4;
385  v3 += XXH_readLE32((const U32*)p, endian) * PRIME32_2; v3 = XXH_rotl32(v3, 13); v3 *= PRIME32_1; p+=4;
386  v4 += XXH_readLE32((const U32*)p, endian) * PRIME32_2; v4 = XXH_rotl32(v4, 13); v4 *= PRIME32_1; p+=4;
387  } while (p<=limit);
388 
389  state->v1 = v1;
390  state->v2 = v2;
391  state->v3 = v3;
392  state->v4 = v4;
393  }
394 
395  if (p < bEnd)
396  {
397  XXH_memcpy(state->memory, p, bEnd-p);
398  state->memsize = (int)(bEnd-p);
399  }
400 
401  return XXH_OK;
402 }
