449  // read data from the source to fill the buffer
450  bool Fill() {
451  unsigned int pos = nSrcPos % vchBuf.size();
452  unsigned int readNow = vchBuf.size() - pos;
453  unsigned int nAvail = vchBuf.size() - (nSrcPos - nReadPos) - nRewind;
454  if (nAvail < readNow)
455  readNow = nAvail;
456  if (readNow == 0)
457  return false;
458  size_t read = fread((void*)&vchBuf[pos], 1, readNow, src);
459  if (read == 0) {
460  throw std::ios_base::failure(feof(src) ? "CBufferedFile::Fill : end of file" : "CBufferedFile::Fill : fread failed");
461  } else {
462  nSrcPos += read;
463  return true;
464  }
