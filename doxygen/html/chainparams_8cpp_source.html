25 // Convert the pnSeeds array into usable address objects.
26 static void convertSeeds(std::vector<CAddress> &vSeedsOut, const unsigned int *data, unsigned int count, int port)
27 {
28  // It'll only connect to one or two seed nodes because once it connects,
29  // it'll get a pile of addresses with newer timestamps.
30  // Seed nodes are given a random 'last seen time' of between one and two
31  // weeks ago.
32  const int64_t nOneWeek = 7*24*60*60;
33  for (unsigned int k = 0; k < count; ++k)
34  {
35  struct in_addr ip;
36  unsigned int i = data[k], t;
37 
38  // -- convert to big endian
39  t = (i & 0x000000ff) << 24u
40  | (i & 0x0000ff00) << 8u
41  | (i & 0x00ff0000) >> 8u
42  | (i & 0xff000000) >> 24u;
43 
44  memcpy(&ip, &t, sizeof(ip));
45 
46  CAddress addr(CService(ip, port));
47  addr.nTime = GetTime()-GetRand(nOneWeek)-nOneWeek;
48  vSeedsOut.push_back(addr);
49  }
50 }
