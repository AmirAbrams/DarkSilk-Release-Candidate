91 CStormnodeDB::ReadResult CStormnodeDB::Read(CStormnodeMan& snodemanToLoad, bool fDryRun)
92 {
93  int64_t nStart = GetTimeMillis();

95  FILE *file = fopen(pathSN.string().c_str(), "rb");
96  CAutoFile filein(file, SER_DISK, CLIENT_VERSION);
97  if (filein.IsNull())
98  {

101  }
102

104  int fileSize = boost::filesystem::file_size(pathSN);
105  int dataSize = fileSize - sizeof(uint256);
106  // Don't try to resize to a negative number if file is small
107  if (dataSize < 0)
108  dataSize = 0;
109  vector<unsigned char> vchData;
110  vchData.resize(dataSize);
111  uint256 hashIn;
112 
113  // read data and checksum from file
114  try {
115  filein.read((char *)&vchData[0], dataSize);
116  filein >> hashIn;
117  }
118  catch (std::exception &e) {
119  error("%s : Deserialize or I/O error - %s", __func__, e.what());
120  return HashReadError;
121  }
122  filein.fclose();
123 
124  CDataStream ssStormnodes(vchData, SER_DISK, CLIENT_VERSION);
125 
126  // verify stored checksum matches input data
127  uint256 hashTmp = Hash(ssStormnodes.begin(), ssStormnodes.end());
128  if (hashIn != hashTmp)
129  {
130  error("%s : Checksum mismatch, data corrupted", __func__);
131  return IncorrectHash;
132  }
133 
134  unsigned char pchMsgTmp[4];
135  std::string strMagicMessageTmp;
136  try {
137  // de-serialize file header (stormnode cache file specific magic message) and ..
138 
139  ssStormnodes >> strMagicMessageTmp;
140 
141  // ... verify the message matches predefined one

145  return IncorrectMagicMessage;

159  }

163  return IncorrectFormat;
164  }
165 
166  LogPrintf("Loaded info from sncache.dat %dms\n", GetTimeMillis() - nStart);
167  LogPrintf(" %s\n", snodemanToLoad.ToString());
168  if(!fDryRun) {
169  LogPrintf("Stormnode manager - cleaning....\n");
170  snodemanToLoad.CheckAndRemove(true);

442 CStormnode* CStormnodeMan::GetNextStormnodeInQueueForPayment(int nBlockHeight, bool fFilterSigTime, int& nCount)
443 {
444  LOCK(cs);
445 
446  CStormnode *pBestStormnode = NULL;
447  std::vector<pair<int64_t, CTxIn> > vecStormnodeLastPaid;
448 
449  /*
450  Make a vector with all of the last paid times
451  */
452 
453  int nSnCount = CountEnabled();
454  BOOST_FOREACH(CStormnode &sn, vStormnodes)
455  {
456  sn.Check();
457  if(!sn.IsEnabled()) continue;
458 
459  // check protocol version
460  if(sn.protocolVersion < stormnodePayments.GetMinStormnodePaymentsProto()) continue;
461 
462  //it's in the list (up to 8 entries ahead of current block to allow propagation) -- so let's skip it
463  if(stormnodePayments.IsScheduled(sn, nBlockHeight)) continue;
464 
465  //it's too new, wait for a cycle
466  if(fFilterSigTime && sn.sigTime + (nSnCount*2.6*60) > GetAdjustedTime()) continue;
467 
468  //make sure it has as many confirmations as there are stormnodes
469  if(sn.GetStormnodeInputAge() < nSnCount) continue;
470 
471  vecStormnodeLastPaid.push_back(make_pair(sn.SecondsSincePayment(), sn.vin));
472  }
473 
474  nCount = (int)vecStormnodeLastPaid.size();
475 
476  //when the network is in the process of upgrading, don't penalize nodes that recently restarted
477  if(fFilterSigTime && nCount < nSnCount/3) return GetNextStormnodeInQueueForPayment(nBlockHeight, false, nCount);
478 
479  // Sort them high to low
480  sort(vecStormnodeLastPaid.rbegin(), vecStormnodeLastPaid.rend(), CompareLastPaid());
481 
482  // Look at 1/10 of the oldest nodes (by last payment), calculate their scores and pay the best one
483  // -- This doesn't look at who is being paid in the +8-10 blocks, allowing for double payments very rarely
484  // -- 1/100 payments should be a double payment on mainnet - (1/(3000/10))*2
485  // -- (chance per block * chances before IsScheduled will fire)
486  int nTenthNetwork = CountEnabled()/10;
487  int nCountTenth = 0;
488  uint256 nHigh = 0;
489  BOOST_FOREACH (PAIRTYPE(int64_t, CTxIn)& s, vecStormnodeLastPaid){
490  CStormnode* psn = Find(s.second);
491  if(!psn) break;
492 
493  uint256 n = psn->CalculateScore(1, nBlockHeight-100);
494  if(n > nHigh){
495  nHigh = n;
496  pBestStormnode = psn;
497  }
498  nCountTenth++;
499  if(nCountTenth >= nTenthNetwork) break;
500  }
501  return pBestStormnode;
502 }

674 void CStormnodeMan::ProcessMessage(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
675 {
676 
677  if(fLiteMode) return; //disable all SandStorm/Stormnode related functionality
678  if(!stormnodeSync.IsBlockchainSynced()) return;
679 
680  LOCK(cs_process_message);
681 
682  if (strCommand == "snb") { //Stormnode Broadcast
683  CStormnodeBroadcast snb;
684  vRecv >> snb;
685 
686  if(mapSeenStormnodeBroadcast.count(snb.GetHash())) { //seen
687  stormnodeSync.AddedStormnodeList(snb.GetHash());
688  return;
689  }
690  mapSeenStormnodeBroadcast.insert(make_pair(snb.GetHash(), snb));
691 
692  int nDoS = 0;
693  if(!snb.CheckAndUpdate(nDoS)){
694 
695  if(nDoS > 0)
696  Misbehaving(pfrom->GetId(), nDoS);
697 
698  //failed
699  return;
700  }
701 
702  // make sure the vout that was signed is related to the transaction that spawned the Stormnode
703  // - this is expensive, so it's only done once per Stormnode
704  if(!sandStormSigner.IsVinAssociatedWithPubkey(snb.vin, snb.pubkey)) {
705  LogPrintf("snb - Got mismatched pubkey and vin\n");
706  Misbehaving(pfrom->GetId(), 33);
707  return;
708  }
709 
710  // make sure it's still unspent
711  // - this is checked later by .check() in many places and by ThreadCheckSandStormPool()
712  if(snb.CheckInputsAndAdd(nDoS)) {
713  // use this as a peer
714  addrman.Add(CAddress(snb.addr), pfrom->addr, 2*60*60);
715  stormnodeSync.AddedStormnodeList(snb.GetHash());
716  } else {
717  LogPrintf("snb - Rejected Stormnode entry %s\n", snb.addr.ToString());
718 
719  if (nDoS > 0)
720  Misbehaving(pfrom->GetId(), nDoS);
721  }
722  }
723 
724  else if (strCommand == "snp") { //Stormnode Ping
725  CStormnodePing snp;
726  vRecv >> snp;
727 
728  LogPrint("stormnode", "snp - Stormnode ping, vin: %s\n", snp.vin.ToString());
729 
730  if(mapSeenStormnodePing.count(snp.GetHash())) return; //seen
731  mapSeenStormnodePing.insert(make_pair(snp.GetHash(), snp));
732 
733  int nDoS = 0;
734  if(snp.CheckAndUpdate(nDoS)) return;
735 
736  if(nDoS > 0) {
737  // if anything significant failed, mark that node
738  Misbehaving(pfrom->GetId(), nDoS);
739  } else {
740  // if nothing significant failed, search existing Stormnode list
741  CStormnode* psn = Find(snp.vin);
742  // if it's known, don't ask for the snb, just return
743  if(psn != NULL) return;
744  }
745 
746  // something significant is broken or sn is unknown,
747  // we might have to ask for a stormnode entry once
748  AskForSN(pfrom, snp.vin);
749 
750  } else if (strCommand == "sseg") { //Get Stormnode list or specific entry
751 
752  CTxIn vin;
753  vRecv >> vin;
754 
755  if(vin == CTxIn()) { //only should ask for this once
756  //local network
757  bool isLocal = (pfrom->addr.IsRFC1918() || pfrom->addr.IsLocal());
758 
759  if(!isLocal && Params().NetworkID() == CChainParams::MAIN) {
760  std::map<CNetAddr, int64_t>::iterator i = mAskedUsForStormnodeList.find(pfrom->addr);
761  if (i != mAskedUsForStormnodeList.end()){
762  int64_t t = (*i).second;
763  if (GetTime() < t) {
764  Misbehaving(pfrom->GetId(), 34);
765  LogPrintf("sseg - peer already asked me for the list\n");
766  return;
767  }
768  }
769  int64_t askAgain = GetTime() + STORMNODES_SSEG_SECONDS;
770  mAskedUsForStormnodeList[pfrom->addr] = askAgain;
771  }
772  } //else, asking for a specific node which is ok
773 
774 
775  int nInvCount = 0;
776 
777  BOOST_FOREACH(CStormnode& sn, vStormnodes) {
778  if(sn.addr.IsRFC1918()) continue; //local network
779 
780  if(sn.IsEnabled()) {
781  LogPrint("stormnode", "sseg - Sending Stormnode entry - %s \n", sn.addr.ToString());
782  if(vin == CTxIn() || vin == sn.vin){
783  CStormnodeBroadcast snb = CStormnodeBroadcast(sn);
784  uint256 hash = snb.GetHash();
785  pfrom->PushInventory(CInv(MSG_STORMNODE_ANNOUNCE, hash));
786  nInvCount++;
787 
788  if(!mapSeenStormnodeBroadcast.count(hash)) mapSeenStormnodeBroadcast.insert(make_pair(hash, snb));
789 
790  if(vin == sn.vin) {
791  LogPrintf("sseg - Sent 1 Stormnode entries to %s\n", pfrom->addr.ToString());
792  return;
793  }
794  }
795  }
796  }
797 
798  if(vin == CTxIn()) {
799  pfrom->PushMessage("ssc", STORMNODE_SYNC_LIST, nInvCount);
800  LogPrintf("sseg - Sent %d Stormnode entries to %s\n", nInvCount, pfrom->addr.ToString());
801  }
802  }
803 }
