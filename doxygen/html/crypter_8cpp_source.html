25 bool CCrypter::SetKeyFromPassphrase(const SecureString& strKeyData, const std::vector<unsigned char>& chSalt, const unsigned int nRounds, const unsigned int nDerivationMethod)
26 {
27  if (nRounds < 1 || chSalt.size() != WALLET_CRYPTO_SALT_SIZE)
28  return false;
29 
30  int i = 0;
31  if (nDerivationMethod == 0)
32  {
33  i = EVP_BytesToKey(EVP_aes_256_cbc(), EVP_sha512(), &chSalt[0],
34  (unsigned char *)&strKeyData[0], strKeyData.size(), nRounds, chKey, chIV);
35  }
36 
37  if (nDerivationMethod == 1)
38  {
39  // Passphrase conversion
40  uint256 scryptHash = scrypt_salted_multiround_hash((const void*)strKeyData.c_str(), strKeyData.size(), &chSalt[0], 8, nRounds);
41 
42  i = EVP_BytesToKey(EVP_aes_256_cbc(), EVP_sha512(), &chSalt[0],
43  (unsigned char *)&scryptHash, sizeof scryptHash, nRounds, chKey, chIV);
44  memory_cleanse(&scryptHash, sizeof scryptHash);
45  }
46 
47 
48  if (i != (int)WALLET_CRYPTO_KEY_SIZE)
49  {
50  memory_cleanse(chKey, sizeof(chKey));
51  memory_cleanse(chIV, sizeof(chIV));
52  return false;
53  }
54

238 bool CCryptoKeyStore::Unlock(const CKeyingMaterial& vMasterKeyIn)
239 {
240  {
241  LOCK(cs_KeyStore);
242  if (!SetCrypted())
243  return false;
244 
245  CryptedKeyMap::const_iterator mi = mapCryptedKeys.begin();
246  for (; mi != mapCryptedKeys.end(); ++mi)
247  {
248  const CPubKey &vchPubKey = (*mi).second.first;
249  const std::vector<unsigned char> &vchCryptedSecret = (*mi).second.second;
250  CKeyingMaterial vchSecret;
251  if(!DecryptSecret(vMasterKeyIn, vchCryptedSecret, vchPubKey.GetHash(), vchSecret))
252  return false;
253  if (vchSecret.size() != 32)
254  return false;
255  CKey key;
256  key.Set(vchSecret.begin(), vchSecret.end(), vchPubKey.IsCompressed());
257  if (key.GetPubKey() == vchPubKey)
258  break;
259  return false;
260  }
261  vMasterKey = vMasterKeyIn;
262  }
263  NotifyStatusChanged(this);
