25 void fill_block(const block *prev_block, const block *ref_block,
26  block *next_block) {
27  block blockR, block_tmp;
28  unsigned i;
29 
30  copy_block(&blockR, ref_block);
31  xor_block(&blockR, prev_block);
32  copy_block(&block_tmp, &blockR);
33 
34  /* Apply Blake2 on columns of 64-bit words: (0,1,...,15) , then
35  (16,17,..31)... finally (112,113,...127) */
36  for (i = 0; i < 8; ++i) {
37  BLAKE2_ROUND_NOMSG(
38  blockR.v[16 * i], blockR.v[16 * i + 1], blockR.v[16 * i + 2],
39  blockR.v[16 * i + 3], blockR.v[16 * i + 4], blockR.v[16 * i + 5],
40  blockR.v[16 * i + 6], blockR.v[16 * i + 7], blockR.v[16 * i + 8],
41  blockR.v[16 * i + 9], blockR.v[16 * i + 10], blockR.v[16 * i + 11],
42  blockR.v[16 * i + 12], blockR.v[16 * i + 13], blockR.v[16 * i + 14],
43  blockR.v[16 * i + 15]);
44  }
45 
46  /* Apply Blake2 on rows of 64-bit words: (0,1,16,17,...112,113), then
47  (2,3,18,19,...,114,115).. finally (14,15,30,31,...,126,127) */
48  for (i = 0; i < 8; i++) {
49  BLAKE2_ROUND_NOMSG(

55  blockR.v[2 * i + 113]);
56  }

92 void fill_segment(const argon2_instance_t *instance,
93  argon2_position_t position) {
94  block *ref_block = NULL, *curr_block = NULL;
95  uint64_t pseudo_rand, ref_index, ref_lane;
96  uint32_t prev_offset, curr_offset;
97  uint32_t starting_index;
98  uint32_t i;
99  int data_independent_addressing;
100  /* Pseudo-random values that determine the reference block position */
101  uint64_t *pseudo_rands = NULL;
102 
103  if (instance == NULL) {
104  return;
105  }
106 
107  data_independent_addressing = (instance->type == Argon2_i);
108 
109  pseudo_rands =
110  (uint64_t *)malloc(sizeof(uint64_t) * (instance->segment_length));
111 
112  if (pseudo_rands == NULL) {
113  return;
114  }
115 
116  if (data_independent_addressing) {
117  generate_addresses(instance, &position, pseudo_rands);
118  }
119 
120  starting_index = 0;
121 
122  if ((0 == position.pass) && (0 == position.slice)) {
123  starting_index = 2; /* we have already generated the first two blocks */
124  }
125 
126  /* Offset of the current block */
127  curr_offset = position.lane * instance->lane_length +
128  position.slice * instance->segment_length + starting_index;
129 
130  if (0 == curr_offset % instance->lane_length) {
131  /* Last block in this lane */
132  prev_offset = curr_offset + instance->lane_length - 1;
133  } else {
134  /* Previous block */
135  prev_offset = curr_offset - 1;
136  }
137 
138  for (i = starting_index; i < instance->segment_length;
139  ++i, ++curr_offset, ++prev_offset) {
140  /*1.1 Rotating prev_offset if needed */
141  if (curr_offset % instance->lane_length == 1) {
142  prev_offset = curr_offset - 1;
143  }
144 
145  /* 1.2 Computing the index of the reference block */
146  /* 1.2.1 Taking pseudo-random value from the previous block */
147  if (data_independent_addressing) {
148  pseudo_rand = pseudo_rands[i];
149  } else {
150  pseudo_rand = instance->memory[prev_offset].v[0];
151  }
152 
153  /* 1.2.2 Computing the lane of the reference block */
154  ref_lane = ((pseudo_rand >> 32)) % instance->lanes;
155 
156  if ((position.pass == 0) && (position.slice == 0)) {
157  /* Can not reference other lanes yet */
158  ref_lane = position.lane;
159  }
160 
161  /* 1.2.3 Computing the number of possible reference block within the
162  * lane.
163  */
164  position.index = i;
165  ref_index = index_alpha(instance, &position, pseudo_rand & 0xFFFFFFFF,
166  ref_lane == position.lane);
167 
168  /* 2 Creating a new block */
169  ref_block =
170  instance->memory + instance->lane_length * ref_lane + ref_index;
171  curr_block = instance->memory + curr_offset;
172  fill_block(instance->memory + prev_offset, ref_block, curr_block);
173  }
174 
175  free(pseudo_rands);
176 }
