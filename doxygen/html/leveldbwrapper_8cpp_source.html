14 void HandleError(const leveldb::Status& status) throw(leveldb_error)
15 {
16  if (status.ok())
17  return;
18  LogPrintf("%s\n", status.ToString());
19  if (status.IsCorruption())
20  throw leveldb_error("Database corrupted");
21  if (status.IsIOError())
22  throw leveldb_error("Database I/O error");
23  if (status.IsNotFound())
24  throw leveldb_error("Database entry missing");
25  throw leveldb_error("Unknown database error");
26 }

28 static leveldb::Options GetOptions(size_t nCacheSize)
29 {
30  leveldb::Options options;
31  options.block_cache = leveldb::NewLRUCache(nCacheSize / 2);
32  options.write_buffer_size = nCacheSize / 4; // up to two write buffers may be held in memory simultaneously
33  options.filter_policy = leveldb::NewBloomFilterPolicy(10);
34  options.compression = leveldb::kNoCompression;
35  options.max_open_files = 64;
36  if (leveldb::kMajorVersion > 1 || (leveldb::kMajorVersion == 1 && leveldb::kMinorVersion >= 16)) {
37  // LevelDB versions before 1.16 consider short writes to be corruption. Only trigger error
38  // on corruption in later versions.
39  options.paranoid_checks = true;
40  }
41  return options;
42 }
