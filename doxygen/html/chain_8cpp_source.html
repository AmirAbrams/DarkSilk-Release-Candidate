96  if (nVersion > CURRENT_VERSION)
97  return DoS(100, error("AcceptBlock() : reject unknown block version %d", nVersion));
98 
99  // Check for duplicate
100  uint256 hash = GetHash();
101  if (mapBlockIndex.count(hash))
102  return error("AcceptBlock() : block already in mapBlockIndex");
103 
104  // Get prev block index
105  map<uint256, CBlockIndex*>::iterator mi = mapBlockIndex.find(hashPrevBlock);
106  if (mi == mapBlockIndex.end())
107  return DoS(10, error("AcceptBlock() : prev block not found"));
108  CBlockIndex* pindexPrev = (*mi).second;
109  int nHeight = pindexPrev->nHeight+1;
110 
111  if (IsProofOfStake() && nHeight < Params().FirstPOSBlock())
112  return DoS(100, error("AcceptBlock() : reject proof-of-stake at height <= %d", nHeight));
113 
114  // Check coinbase timestamp
115  if (GetBlockTime() > FutureDrift((int64_t)vtx[0].nTime, IsProofOfStake()))
116  return DoS(50, error("AcceptBlock() : coinbase timestamp is too early"));
117 
118  // Check coinstake timestamp
119  if (IsProofOfStake() && !CheckCoinStakeTimestamp(nHeight, GetBlockTime(), (int64_t)vtx[1].nTime))
120  return DoS(50, error("AcceptBlock() : coinstake timestamp violation nTimeBlock=%d nTimeTx=%u", GetBlockTime(), vtx[1].nTime));
121 
122  // Check proof-of-work or proof-of-stake
123  if (nBits != GetNextTargetRequired(pindexPrev, IsProofOfStake()))
124  return DoS(100, error("AcceptBlock() : incorrect %s", IsProofOfWork() ? "proof-of-work" : "proof-of-stake"));
125 
126  // Check timestamp against prev
127  LogPrintf("GetBlockTime(): %d, <=? pindexPrev->GetPastTimeLimit(): %d\nFutureDrift(GetBlockTime()): %d, <?pindexPrev->GetBlockTime(): %d\n",GetBlockTime(),pindexPrev->GetPastTimeLimit(),FutureDrift(GetBlockTime()),pindexPrev->GetBlockTime());
128  if (GetBlockTime() <= pindexPrev->GetPastTimeLimit() || FutureDrift(GetBlockTime(), IsProofOfStake()) < pindexPrev->GetBlockTime())
129  return error("AcceptBlock() : block's timestamp is too early");
130 
131  // Check that all transactions are finalized
132  BOOST_FOREACH(const CTransaction& tx, vtx)
133  if (!IsFinalTx(tx, nHeight, GetBlockTime()))
134  return DoS(10, error("AcceptBlock() : contains a non-final transaction"));
135 
136  // Check that the block chain matches the known block chain up to a checkpoint
137  if (!Checkpoints::CheckHardened(nHeight, hash))
138  return DoS(100, error("AcceptBlock() : rejected by hardened checkpoint lock-in at %d", nHeight));
139 
140  uint256 hashProof;
141  // Verify hash target and signature of coinstake tx
142  if (IsProofOfStake())
143  {
144  uint256 targetProofOfStake;
145  if (!CheckProofOfStake(pindexPrev, vtx[1], nBits, hashProof, targetProofOfStake))
146  {
147  return error("AcceptBlock() : check proof-of-stake failed for block %s", hash.ToString());
148  }
149  }
150  // PoW is checked in CheckBlock()
151  if (IsProofOfWork())
152  {
153  hashProof = GetPoWHash();
154  }
155 
156  // Check that the block satisfies synchronized checkpoint
157  if (!Checkpoints::CheckSync(nHeight))
158  return error("AcceptBlock() : rejected by synchronized checkpoint");
159 
160  // Enforce rule that the coinbase starts with serialized block height
161  CScript expect = CScript() << nHeight;
162  if (vtx[0].vin[0].scriptSig.size() < expect.size() ||
163  !std::equal(expect.begin(), expect.end(), vtx[0].vin[0].scriptSig.begin()))
164  return DoS(100, error("AcceptBlock() : block height mismatch in coinbase"));
165 
166  // Write block to history file
167  if (!CheckDiskSpace(::GetSerializeSize(*this, SER_DISK, CLIENT_VERSION)))
168  return error("AcceptBlock() : out of disk space");
169 
170  unsigned int nFile = -1;
171  unsigned int nBlockPos = 0;
172  if (!WriteToDisk(nFile, nBlockPos))
173  return error("AcceptBlock() : WriteToDisk failed");
174 
175  if (!AddToBlockIndex(nFile, nBlockPos, hashProof))
176  return error("AcceptBlock() : AddToBlockIndex failed");
177 
178  // Relay inventory, but don't relay old inventory during initial block download
179  int nBlockEstimate = Checkpoints::GetTotalBlocksEstimate();
180  if (hashBestChain == hash)
181  {
182  LOCK(cs_vNodes);
183  BOOST_FOREACH(CNode* pnode, vNodes)
184  if (nBestHeight > (pnode->nStartingHeight != -1 ? pnode->nStartingHeight - 2000 : nBlockEstimate))
185  pnode->PushInventory(CInv(MSG_BLOCK, hash));
186  }

216 bool static BuildAddrIndex(const CScript &script, std::vector<uint160>& addrIds)
217 {
218  CScript::const_iterator pc = script.begin();
219  CScript::const_iterator pend = script.end();
220  std::vector<unsigned char> data;
221  opcodetype opcode;
222  bool fHaveData = false;
223  while (pc < pend) {
224  script.GetOp(pc, opcode, data);
225  if (0 <= opcode && opcode <= OP_PUSHDATA4 && data.size() >= 8) { // data element
226  uint160 addrid = 0;
227  if (data.size() <= 20) {
228  memcpy(&addrid, &data[0], data.size());
229  } else {
230  addrid = Hash160(data);
231  }
232  addrIds.push_back(addrid);
233  fHaveData = true;
234  }
235  }
236  if (!fHaveData) {
237  uint160 addrid = Hash160(script);
238  addrIds.push_back(addrid);
239  return true;
240  }
241  else
242  {
243  if(addrIds.size() > 0)
244  return true;
245  else
246  return false;
247  }

514 bool Reorganize(CTxDB& txdb, CBlockIndex* pindexNew)
515 {
516  LogPrintf("REORGANIZE\n");
517 
518  // Find the fork
519  CBlockIndex* pfork = pindexBest;
520  CBlockIndex* plonger = pindexNew;
521  while (pfork != plonger)
522  {
523  while (plonger->nHeight > pfork->nHeight)
524  if (!(plonger = plonger->pprev))
525  return error("Reorganize() : plonger->pprev is null");
526  if (pfork == plonger)
527  break;
528  if (!(pfork = pfork->pprev))
529  return error("Reorganize() : pfork->pprev is null");
530  }
531 
532  // List of what to disconnect
533  vector<CBlockIndex*> vDisconnect;
534  for (CBlockIndex* pindex = pindexBest; pindex != pfork; pindex = pindex->pprev)
535  vDisconnect.push_back(pindex);
536 
537  // List of what to connect
538  vector<CBlockIndex*> vConnect;
539  for (CBlockIndex* pindex = pindexNew; pindex != pfork; pindex = pindex->pprev)
540  vConnect.push_back(pindex);
541  reverse(vConnect.begin(), vConnect.end());
542 
543  LogPrintf("REORGANIZE: Disconnect %u blocks; %s..%s\n", vDisconnect.size(), pfork->GetBlockHash().ToString(), pindexBest->GetBlockHash().ToString());
544  LogPrintf("REORGANIZE: Connect %u blocks; %s..%s\n", vConnect.size(), pfork->GetBlockHash().ToString(), pindexNew->GetBlockHash().ToString());
545 
546  // Disconnect shorter branch
547  list<CTransaction> vResurrect;
548  BOOST_FOREACH(CBlockIndex* pindex, vDisconnect)
549  {
550  CBlock block;
551  if (!block.ReadFromDisk(pindex))
552  return error("Reorganize() : ReadFromDisk for disconnect failed");
553  if (!block.DisconnectBlock(txdb, pindex))
554  return error("Reorganize() : DisconnectBlock %s failed", pindex->GetBlockHash().ToString());
555 
556  // Queue memory transactions to resurrect.
557  // We only do this for blocks after the last checkpoint (reorganisation before that
558  // point should only happen with -reindex/-loadblock, or a misbehaving peer.
559  BOOST_REVERSE_FOREACH(const CTransaction& tx, block.vtx)
560  if (!(tx.IsCoinBase() || tx.IsCoinStake()) && pindex->nHeight > Checkpoints::GetTotalBlocksEstimate())
561  vResurrect.push_front(tx);
562  }
563 
564  // Connect longer branch
565  CValidationState state;
566  vector<CTransaction> vDelete;
567  for (unsigned int i = 0; i < vConnect.size(); i++)
568  {
569  CBlockIndex* pindex = vConnect[i];
570  CBlock block;
571  if (!block.ReadFromDisk(pindex))
572  return error("Reorganize() : ReadFromDisk for connect failed");
573  if (!block.ConnectBlock(txdb, pindex))
574  {
575  // Invalid block
576  return error("Reorganize() : ConnectBlock %s failed", pindex->GetBlockHash().ToString());
577  }
578 
579  // Queue memory transactions to delete
580  BOOST_FOREACH(const CTransaction& tx, block.vtx)
581  vDelete.push_back(tx);
582  }
583  if (!txdb.WriteHashBestChain(pindexNew->GetBlockHash()))
584  return error("Reorganize() : WriteHashBestChain failed");

603

605  BOOST_FOREACH(CTransaction& tx, vDelete) {

608  }

678 bool CBlock::CheckBlock(CValidationState& state, bool fCheckPOW, bool fCheckMerkleRoot, bool fCheckSig)
679 {
680  // These are checks that are independent of context
681  // that can be verified before saving an orphan block.
682 
683  // Size limits
684  if (vtx.empty() || vtx.size() > MAX_BLOCK_SIZE || ::GetSerializeSize(*this, SER_NETWORK, PROTOCOL_VERSION) > MAX_BLOCK_SIZE)
685  return DoS(100, error("CheckBlock() : size limits failed"));
686 
687  // Check proof of work matches claimed amount
688  if (fCheckPOW && IsProofOfWork() && !CheckProofOfWork(GetPoWHash(), nBits))
689  return DoS(50, error("CheckBlock() : proof of work failed"));
690 
691  // Check timestamp
692  if (GetBlockTime() > FutureDrift(GetAdjustedTime(), IsProofOfStake()))
693  return error("CheckBlock() : block timestamp too far in the future");
694 
695  // First transaction must be coinbase, the rest must not be
696  if (vtx.empty() || !vtx[0].IsCoinBase())
697  return DoS(100, error("CheckBlock() : first tx is not coinbase"));
698  for (unsigned int i = 1; i < vtx.size(); i++)
699  if (vtx[i].IsCoinBase())
700  return DoS(100, error("CheckBlock() : more than one coinbase"));
701 
702  if (IsProofOfStake())
703  {
704  // Coinbase output should be empty if proof-of-stake block
705  if (vtx[0].vout.size() != 1 || !vtx[0].vout[0].IsEmpty())
706  return DoS(100, error("CheckBlock() : coinbase output not empty for proof-of-stake block"));
707 
708  // Second transaction must be coinstake, the rest must not be
709  if (vtx.empty() || !vtx[1].IsCoinStake())
710  return DoS(100, error("CheckBlock() : second tx is not coinstake"));
711  for (unsigned int i = 2; i < vtx.size(); i++)
712  if (vtx[i].IsCoinStake())
713  return DoS(100, error("CheckBlock() : more than one coinstake"));
714  }
715 
716  // Check proof-of-stake block signature
717  if (fCheckSig && !CheckBlockSignature())
718  return DoS(100, error("CheckBlock() : bad proof-of-stake block signature"));
719 
720 
721 // ----------- instantX transaction scanning -----------
722 
723  if(IsSporkActive(SPORK_3_INSTANTX_BLOCK_FILTERING)){
724  BOOST_FOREACH(const CTransaction& tx, vtx){
725  if (!tx.IsCoinBase()){
726  //only reject blocks when it's based on complete consensus
727  BOOST_FOREACH(const CTxIn& in, tx.vin){
728  if(mapLockedInputs.count(in.prevout)){
729  if(mapLockedInputs[in.prevout] != tx.GetHash()){
730  if(fDebug) { LogPrintf("CheckBlock() : found conflicting transaction with transaction lock %s %s\n", mapLockedInputs[in.prevout].ToString().c_str(), tx.GetHash().ToString().c_str()); }
731  return DoS(0, error("CheckBlock() : found conflicting transaction with transaction lock"));
732  }
733  }
734  }
735  }
736  }
737  } else {
738  LogPrintf("CheckBlock() : skipping transaction locking checks\n");
739  }
740 
741 
742  // ----------- stormnode payments -----------
743 
744  CBlockIndex *pindexPrev = pindexBest;
745 
746  if(pindexPrev == NULL)
747  {
748  //TODO (AA): Put these back once primitives are completed
749  //if(IsBlockPayeeValid(block.vtx[0], pindexPrev->nHeight+1))
750  //{
751  // return DoS(100, error("CheckBlock() : Couldn't find stormnode payment or payee"));
752  //}
753  }
754 
755  // Check transactions
756  BOOST_FOREACH(const CTransaction& tx, vtx){
757  if (!tx.CheckTransaction(state))
758  return DoS(tx.nDoS, error("CheckBlock() : CheckTransaction failed"));
759 
760  // ppcoin: check transaction timestamp
761  if (GetBlockTime() < (int64_t)tx.nTime)
762  return DoS(50, error("CheckBlock() : block timestamp earlier than transaction timestamp"));
763  }
764 
765  // Check for duplicate txids. This is caught by ConnectInputs(),
766  // but catching it earlier avoids a potential DoS attack:
767  set<uint256> uniqueTx;
768  BOOST_FOREACH(const CTransaction& tx, vtx)
769  {
770  uniqueTx.insert(tx.GetHash());
771  }
772  if (uniqueTx.size() != vtx.size())
773  return DoS(100, error("CheckBlock() : duplicate transaction"));
774 
775  unsigned int nSigOps = 0;
776  BOOST_FOREACH(const CTransaction& tx, vtx)
777  {
778  nSigOps += GetLegacySigOpCount(tx);
779  }
780  if (nSigOps > MAX_BLOCK_SIGOPS)
781  return DoS(100, error("CheckBlock() : out-of-bounds SigOpCount"));
782 
783  // Check merkle root
784  if (fCheckMerkleRoot && hashMerkleRoot != BuildMerkleTree())
785  return DoS(100, error("CheckBlock() : hashMerkleRoot mismatch"));
786 
787 
788  return true;
789 }
