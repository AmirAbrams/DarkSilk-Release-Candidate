66 CStormnodePaymentDB::ReadResult CStormnodePaymentDB::Read(CStormnodePayments& objToLoad, bool fDryRun)
67 {
68 
69  int64_t nStart = GetTimeMillis();
70  // open input file, and associate with CAutoFile
71  FILE *file = fopen(pathDB.string().c_str(), "rb");
72  CAutoFile filein(file, SER_DISK, CLIENT_VERSION);
73  if (filein.IsNull())
74  {
75  error("%s : Failed to open file %s", __func__, pathDB.string());
76  return FileError;
77  }
78 
79  // use file size to size memory buffer
80  int fileSize = boost::filesystem::file_size(pathDB);
81  int dataSize = fileSize - sizeof(uint256);
82  // Don't try to resize to a negative number if file is small
83  if (dataSize < 0)
84  dataSize = 0;
85  vector<unsigned char> vchData;
86  vchData.resize(dataSize);
87  uint256 hashIn;
88 
89  // read data and checksum from file
90  try {
91  filein.read((char *)&vchData[0], dataSize);
92  filein >> hashIn;
93  }
94  catch (std::exception &e) {
95  error("%s : Deserialize or I/O error - %s", __func__, e.what());
96  return HashReadError;
97  }
98  filein.fclose();
99 
100  CDataStream ssObj(vchData, SER_DISK, CLIENT_VERSION);
101 
102  // verify stored checksum matches input data
103  uint256 hashTmp = Hash(ssObj.begin(), ssObj.end());
104  if (hashIn != hashTmp)
105  {
106  error("%s : Checksum mismatch, data corrupted", __func__);
107  return IncorrectHash;
108  }
109 
110 
111  unsigned char pchMsgTmp[4];
112  std::string strMagicMessageTmp;
113  try {
114  // de-serialize file header (stormnode cache file specific magic message) and ..
115  ssObj >> strMagicMessageTmp;
116 
117  // ... verify the message matches predefined one
118  if (strMagicMessage != strMagicMessageTmp)
119  {
120  error("%s : Invalid stormnode payement cache magic message", __func__);
121  return IncorrectMagicMessage;
122  }
123 
124 
125  // de-serialize file header (network specific magic number) and ..
126  ssObj >> FLATDATA(pchMsgTmp);
127 
128  // ... verify the network matches ours
129  if (memcmp(pchMsgTmp, Params().MessageStart(), sizeof(pchMsgTmp)))
130  {
131  error("%s : Invalid network magic number", __func__);
132  return IncorrectMagicNumber;
133  }
134 
135  // de-serialize data into CStormnodePayments object
136  ssObj >> objToLoad;
137  }
138  catch (std::exception &e) {
139  objToLoad.Clear();
140  error("%s : Deserialize or I/O error - %s", __func__, e.what());
141  return IncorrectFormat;
142  }
143 
144  LogPrintf("Loaded info from snpayments.dat %dms\n", GetTimeMillis() - nStart);
145  LogPrintf(" %s\n", objToLoad.ToString());
146  if(!fDryRun) {
147  LogPrintf("Stormnode payments manager - cleaning....\n");
148  objToLoad.CleanPaymentList();
149  LogPrintf("Stormnode payments manager - result:\n");
150  LogPrintf(" %s\n", objToLoad.ToString());
151  }
152 
153  return Ok;
154 }
